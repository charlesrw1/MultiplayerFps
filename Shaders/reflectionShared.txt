

layout(binding = 7) uniform sampler2D PBR_brdflut;
layout(binding = 8) uniform samplerCubeArray probes;
layout(binding = 9) uniform samplerCube skyprobe;
uniform int lum_adjust_mode = 4;	// 4 looks the best
uniform int num_cubemaps = 0;
struct R_ReflectionVolume {
	mat4 transform;
	vec4 size;
	vec4 bounds_min;
	vec4 bounds_max;
};

layout(std430, binding =11) readonly buffer BestCubemapForProbe
{
	R_ReflectionVolume data[];
} Cubemaps;


// forget where this is from, frostbite paper i think
float compute_specular_occlusion(float ao, float NdotV, float roughness)
{
    float out_val = (pow(NdotV + ao, exp2(-16.0 * roughness - 1.0)) - 1.0 + ao);
	return clamp(out_val,0,1);
}

// p = test_pos - box_pos, b = box_size
float sdBox( vec3 p, vec3 b )
{
	  vec3 q = abs(p) - b;
	  return length(max(q,0.0)) + min(max(q.x,max(q.y,q.z)),0.0);
}

float luminance(vec3 v)
{
    return dot(v, vec3(0.2126, 0.7152, 0.0722));
}



vec3 get_cubemap_volume_index_debug(vec3 worldpos, vec3 outirrad)
{

	float in_lum = luminance(outirrad);

	float weight_sum = 0.0;
	vec3 sum = vec3(0.0);
			const float MAX_REFLECTION_LOD = 7.0;
	float max_priority = -1000;
	vec3 fade_out_color = vec3(1,0,1);
	for(int i=0;i<num_cubemaps;i++) {
		mat4 transform = Cubemaps.data[i].transform;
		vec4 size = Cubemaps.data[i].size;
		vec4 bmin = Cubemaps.data[i].bounds_min;
		vec4 bmax = Cubemaps.data[i].bounds_max;
		float priority = size.w;
		if(is_point_in_aabb(worldpos,bmin.xyz,bmax.xyz)) {
			vec3 middle = (bmax.xyz+bmin.xyz)*0.5;
			vec3 size = (bmax.xyz-bmin.xyz);
			
		
			float dist = sdBox(worldpos-middle, size*0.5);
			float alpha = 1.0-smoothstep(1.0,0.0,abs(dist));
			
			max_priority = max(max_priority,priority);
			
			
			//vec3 pf = vec3(min(1,in_lum/bmin.w));// randColor(i);
			vec3 pf = randColor(i);
			if(priority<max_priority) {
				fade_out_color = pf;
				break;
			}
			
			sum +=pf*alpha;
			weight_sum += alpha;	
		}
	}
	if(weight_sum <= 1.0) {
		sum += fade_out_color*(1.0-weight_sum);
		weight_sum = 1.0;
	}
	return sum / weight_sum;
}

vec3 get_prefiltered_color(vec3 worldpos, vec3 R, float Roughness,float nDotV, vec3 outirrad)
{
	float weight_sum = 0.0;
	vec3 sum = vec3(0.0);
			const float MAX_REFLECTION_LOD = 7.0;

	float in_lum = luminance(outirrad);

	float max_priority = -1000.0;
	vec3 fade_out_color = vec3(-1);
	for(int i=0;i<num_cubemaps;i++) {
		mat4 transform = Cubemaps.data[i].transform;
		vec4 size = Cubemaps.data[i].size;
		vec4 bmin = Cubemaps.data[i].bounds_min;
		vec4 bmax = Cubemaps.data[i].bounds_max;
		float priority = size.w;

		
		if(is_point_in_aabb(worldpos,bmin.xyz,bmax.xyz)) {
		
			vec3 middle = (bmax.xyz+bmin.xyz)*0.5;
			vec3 size = (bmax.xyz-bmin.xyz);
			
		
			float dist = sdBox(worldpos-middle, size*0.5);
			float alpha = 1.0-smoothstep(1.0,0.0,abs(dist));
			
		
			// if blending out the seams, then you can pick a lower priroty box. ie a blend out from a higher to a lower priority
			// note, volumes already sorted from highest to lowest priority.
			max_priority = max(max_priority,priority);

	
			vec3 pf = textureLod(probes,vec4(R,float(i)),Roughness*MAX_REFLECTION_LOD).rgb;
			// scale the brightness down. bmin.w is precomputed luminance at the position of the specular probe see probeLumCalc_C.txt
			float brightness_adjust = min(1,in_lum/bmin.w);
			if(lum_adjust_mode==0)
				brightness_adjust=1.0;
			else if(lum_adjust_mode==2) {
				brightness_adjust = mix(1,brightness_adjust,Roughness);
			}
			else if(lum_adjust_mode==3) {
				brightness_adjust = mix(1,brightness_adjust,pow(Roughness,0.5));
			}	
			else if(lum_adjust_mode==4) {
				// from treyarch presentation
				// wtf magic numbers
				float mip = Roughness*MAX_REFLECTION_LOD;
				float max_spec = max(
					1.26816,
					max(
						9.13681*exp2(6.85741-2*mip)*nDotV,
						9.70809*exp2(7.085-mip-0.403181*mip*mip)*nDotV
					));
				float adjust_max = max_spec*in_lum;
				brightness_adjust = adjust_max/(adjust_max+bmin.w);
			
			}
			
			pf *= brightness_adjust;
			
			if(priority<max_priority) {
				fade_out_color = pf;
				break;
			}
	
			sum +=pf*alpha;
			weight_sum += alpha;
		}
	}
	
	if(weight_sum < 1.0) {
		if(fade_out_color.x<0)
			fade_out_color = textureLod(skyprobe,vec3(R),Roughness*MAX_REFLECTION_LOD).rgb;
		sum += fade_out_color*(1.0-weight_sum);
		weight_sum = 1.0;
	}
	return sum / weight_sum;
}

vec3 pbr_calculate_reflection(vec3 worldspace, vec3 Albedo, float Roughness, float Metalness, vec3 Normal, float ambient_occlusion, vec3 ref_irrad)
{
	vec3 view_pos = g.viewpos_time.xyz;
	vec3 view_front = g.viewfront.xyz;

	vec3 V = normalize(view_pos-worldspace);
	vec3 R = reflect(-V,Normal);
	
	vec3 F0 = vec3(0.04);
	F0 = mix(F0,Albedo,Metalness);
	
	const float MAX_REFLECTION_LOD = 7.0;
	//vec3 prefiltered_color = textureLod(skyprobe,vec3(R), Roughness*MAX_REFLECTION_LOD).rgb;
	//vec3 prefiltered_color = pow(ref_irrad,vec3(1.5));
	float NdotV = max(dot(Normal,V),0.0);
	
	vec3 prefiltered_color = get_prefiltered_color(worldspace+Normal*0.3,R,Roughness, NdotV,ref_irrad);///get_prefiltered_color(worldspace,R,1.0);// textureLod(probes,vec4(R,float(probe_num)),Roughness*MAX_REFLECTION_LOD).rgb;
	//prefiltered_color *= ref_irrad;

	
	
	vec3 F = FresnelSchlickRougness(NdotV,F0,Roughness);
	
	
	//float F90 = clamp(50*dot(F0,vec3(0.33)),0,1);
	//vec3 F = modified_f_schlick(F0,F90,NdotV);
	
	vec2 env_brdf = texture(PBR_brdflut,vec2(max(dot(Normal,V),0.0),Roughness)).rg;
	
	
	vec3 specular = prefiltered_color*(F*env_brdf.x+env_brdf.y);
	float spec_occlusion =compute_specular_occlusion(ambient_occlusion,NdotV,Roughness);
	

	return specular*spec_occlusion;
}
