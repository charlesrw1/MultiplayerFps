#include "ShaderBufferShared.txt"

layout(location = 0) out vec4 FS_OUT_Color;

layout(binding = 0) uniform sampler2D probe_texture;
layout(location = 0) in vec3 FS_IN_Fragpos;
layout(location = 1) in vec3 FS_IN_Normal;
layout(location = 2) flat in ivec4 FS_IN_ProbeCoord;
layout(std140, binding = 16) uniform MeshDebugProbeFragPushConstsUbo { MeshDebugProbeFragPushConsts pcf; };

// File-scope declarations referenced by ddgiShared.txt's `calculate_irradiance`
// (not called here, but the function still needs the identifiers to resolve).
ivec3 vol_grid;
int   vol_offset;

#include "ddgiShared.txt"

void main()
{
    vec2 uv       = OctahedralEncode(normalize(FS_IN_Normal));
    vec2 final_uv = get_irrad_probe_uv(uv, FS_IN_ProbeCoord.xyz, pcf.vol_grid.xyz, pcf.vol_offset);
    vec3 outc     = texture(probe_texture, final_uv).rgb;
    FS_OUT_Color  = vec4(pow(outc, vec3(1.0 / 2.2)), 1.0);
}
