vec3 pbr_calculate_reflection2(vec3 Albedo, float Roughness, float Metalness, vec3 Normal, vec3 V, vec3 prefiltered)
{
        vec3 R = reflect(-V,Normal);
vec3 F0 = vec3(0.04);
        F0 = mix(F0,Albedo,Metalness);

        //vec3 prefiltered_color = textureLod(skyprobe,vec3(R), Roughness*MAX_REFLECTION_LOD).rgb;
        //vec3 prefiltered_color = pow(ref_irrad,vec3(1.5));
        float NdotV = max(dot(Normal,V),0.0);

        vec3 prefiltered_color = prefiltered;// textureLod(probes,vec4(R,float(probe_num)),Roughness*MAX_REFLECTION_LOD).rgb;
        //prefiltered_color *= ref_irrad;



        vec3 F = FresnelSchlickRougness(NdotV,F0,Roughness);


        //float F90 = clamp(50*dot(F0,vec3(0.33)),0,1);
        //vec3 F = modified_f_schlick(F0,F90,NdotV);

        vec2 env_brdf = texture(pbrLUT,vec2(max(dot(Normal,V),0.0),Roughness)).rg;


        vec3 specular = prefiltered_color*(F*env_brdf.x+env_brdf.y);
     
        return specular;
}
