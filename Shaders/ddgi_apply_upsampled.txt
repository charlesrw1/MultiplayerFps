

layout(location = 0) out vec4 FS_OUT_Color;

#include "SharedGpuTypes.txt"

layout(binding = 0) uniform sampler2D GBUFFER0;		// ws normal (xyz)
layout(binding = 1) uniform sampler2D GBUFFER1;		// albedo
layout(binding = 2) uniform sampler2D GBUFFER2;		// metallic,roughness, custom, matid, custom, matid
layout(binding = 3) uniform sampler2D DEPTH;		// depth texture (writes disabled)

layout(binding = 4) uniform sampler2D SSAOBlurred;
layout(binding = 5) uniform sampler2D ddgi_accum;

layout(binding = 6) uniform sampler2D ssr_accum;



layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};

#include "commonUtil.txt"
#include "LightingShared.txt"
#include "pbrShared.txt"
#include "ddgiShared.txt"
#include "reflectionShared.txt"

uniform float specular_ao_intensity = 3.0;

#if 1
bool is_ssr_texel_valid(vec3 ssrin)
{
	return ssrin.r + ssrin.g + ssrin.b >= 1e-2;
}
vec3 pbr_calculate_reflection_for_ssr(vec3 Albedo, float Roughness, float Metalness, vec3 Normal, vec3 V, vec3 prefiltered)
{
        vec3 R = reflect(-V,Normal);
		vec3 F0 = vec3(0.04);
        F0 = mix(F0,Albedo,Metalness);

        //vec3 prefiltered_color = textureLod(skyprobe,vec3(R), Roughness*MAX_REFLECTION_LOD).rgb;
        //vec3 prefiltered_color = pow(ref_irrad,vec3(1.5));
        float NdotV = max(dot(Normal,V),0.0);

        vec3 prefiltered_color = prefiltered;// textureLod(probes,vec4(R,float(probe_num)),Roughness*MAX_REFLECTION_LOD).rgb;
        //prefiltered_color *= ref_irrad;



        vec3 F = FresnelSchlickRougness(NdotV,F0,Roughness);


        //float F90 = clamp(50*dot(F0,vec3(0.33)),0,1);
        //vec3 F = modified_f_schlick(F0,F90,NdotV);

        vec2 env_brdf = texture(PBR_brdflut,vec2(max(dot(Normal,V),0.0),Roughness)).rg;


        vec3 specular = prefiltered_color*(F*env_brdf.x+env_brdf.y);
     
        return specular;
}

uniform vec2 ssr_lum_range = vec2(0.0,0.1);

void main()
{
	vec2 uv = (vec2(gl_FragCoord.xy)) / vec2(g.viewport_size.xy);
	
	ivec2 texel_to_fetch = ivec2(gl_FragCoord.xy);

	vec4 depth_tex = texelFetch(DEPTH,texel_to_fetch,0);
	vec4 normal_t = texelFetch(GBUFFER0,texel_to_fetch,0);
	vec4 metal_rough = texelFetch(GBUFFER2,texel_to_fetch,0);
	vec4 albedo = texelFetch(GBUFFER1,texel_to_fetch,0);
	
	uint type = get_gbuffer_material_type(metal_rough);
	if(type==MATERIAL_TYPE_UNLIT)
		discard;
		
	const vec3 direction = normal_t.xyz;
	const vec3 worldspace = WorldPosFromDepth(uv, depth_tex.r);
	const float ssao_value = texture(SSAOBlurred, uv).r;

	const vec3 outirrad = texture(ddgi_accum,uv).rgb;
	vec3 rad = vec3(0.0);
	rad = outirrad * pbr_get_albedo_mult(worldspace, albedo.rgb, metal_rough.y, metal_rough.x, direction) * ssao_value;
	vec3 spec_color = vec3(0);
	const float spec_ao = pow(ssao_value,specular_ao_intensity);
	vec3 cubemap_color = pbr_calculate_reflection(worldspace, albedo.xyz, metal_rough.y, metal_rough.x, normal_t.xyz,spec_ao,outirrad);
	const float lum_of_cubemaps = luminance(cubemap_color);
	// never darken
	vec3 ssr_color = texture(ssr_accum,uv).rgb;
	if(is_ssr_texel_valid(ssr_color)) {
		const vec3 V = -normalize(worldspace-g.viewpos_time.xyz);
		spec_color = pbr_calculate_reflection_for_ssr(albedo.rgb, metal_rough.y,metal_rough.x,direction,V,ssr_color);
		const float NdotV = max(dot(direction,V),0.0);
		float spec_occlusion =compute_specular_occlusion(spec_ao,NdotV,metal_rough.g);
		const float lum_of_ssr = luminance(ssr_color)*spec_occlusion;
		// hack on hack but looks "ok"
		spec_color = mix(cubemap_color,spec_color, smoothstep(lum_of_cubemaps+ssr_lum_range.x, lum_of_cubemaps+ssr_lum_range.y,lum_of_ssr));
	}
	else {
		spec_color = cubemap_color;
	}
	FS_OUT_Color = vec4(rad+spec_color,1);
}
#endif
