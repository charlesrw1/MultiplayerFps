

layout(location = 0) out vec4 FS_OUT_Color;

#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"

layout (binding = 7, std140) uniform DdgiRuntimeParamsUbo { DdgiRuntimeParams params; };

layout(binding = 0) uniform sampler2D GBUFFER0;		// ws normal (xyz)
layout(binding = 1) uniform sampler2D GBUFFER1;		// albedo
layout(binding = 2) uniform sampler2D GBUFFER2;		// metallic,roughness, custom, matid, custom, matid
layout(binding = 3) uniform sampler2D DEPTH;		// depth texture (writes disabled)

layout(binding = 4) uniform sampler2D SSAOBlurred;
layout(binding = 5) uniform sampler2D ddgi_accum;

layout(binding = 6) uniform sampler2D ssr_accum;



layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};

#include "commonUtil.txt"
#include "LightingShared.txt"
#include "pbrShared.txt"
#include "pbrModels.txt"
#include "ddgiShared.txt"
#include "reflectionShared.txt"

#define specular_ao_intensity params.specular_ao_intensity

void main()
{
	vec2 uv = (vec2(gl_FragCoord.xy)) / vec2(g.viewport_size.xy);
	ivec2 texel_to_fetch = ivec2(gl_FragCoord.xy);

	vec4 depth_tex = texelFetch(DEPTH, texel_to_fetch, 0);
	vec4 normal_t = texelFetch(GBUFFER0, texel_to_fetch, 0);
	vec4 metal_rough = texelFetch(GBUFFER2, texel_to_fetch, 0);
	vec4 albedo = texelFetch(GBUFFER1, texel_to_fetch, 0);

	uint type = get_gbuffer_material_type(metal_rough);
	if (type == MATERIAL_TYPE_UNLIT)
		discard;

	const vec3 direction = normal_t.xyz;
	const vec3 worldspace = WorldPosFromDepth(uv, depth_tex.r);
	const float ssao_value = texture(SSAOBlurred, uv).r;

	const vec3 outirrad = texture(ddgi_accum, uv).rgb;
	float alias0 = get_gbuffer_alias0(metal_rough);
	float alias1 = get_gbuffer_alias1(albedo);
	vec3 rad = outirrad * pbr_get_albedo_mult(worldspace, albedo.rgb, metal_rough.y, metal_rough.x, direction) * ssao_value;
	rad *= apply_material_diffuse_atten(type, direction, worldspace, alias0, alias1);

	const vec3 V = normalize(g.viewpos_time.xyz - worldspace);
	const float NdotV = max(dot(direction, V), 0.0);
	const float spec_ao = pow(ssao_value, specular_ao_intensity);
	const float spec_occlusion = compute_specular_occlusion(spec_ao, NdotV, metal_rough.g);

	// Raw cubemap color (before BRDF weighting)
	vec3 R = reflect(-V, direction);
	vec3 cubemap_raw = get_prefiltered_color(worldspace + direction * 0.3, R, metal_rough.y, NdotV, outirrad);

	// SSR: alpha channel carries hit confidence (0=miss, 1=full hit)
	vec4 ssr = texture(ssr_accum, uv);
	vec3 reflections = mix(cubemap_raw, ssr.rgb, clamp(ssr.a, 0.0, 1.0));

	// Apply specular BRDF weighting once to the mixed result
	vec3 F0 = mix(vec3(0.04), albedo.rgb, metal_rough.x);
	vec3 F = FresnelSchlickRougness(NdotV, F0, metal_rough.y);
	if (type == MATERIAL_TYPE_IRIDESCENCE) {
		vec3 irid_tint = thin_film_iridescence(NdotV, alias0);
		F *= mix(vec3(1.0), irid_tint * 2.0, alias1);
	}
	vec2 env_brdf = texture(PBR_brdflut, vec2(NdotV, metal_rough.y)).rg;
	vec3 spec_color = reflections * (F * env_brdf.x + env_brdf.y) * spec_occlusion;

	if (type == MATERIAL_TYPE_CLEARCOAT) {
		float cc_intensity = alias0;
		float cc_roughness = alias1;
		vec3 F_cc = FresnelSchlickRougness(NdotV, vec3(0.04), cc_roughness);
		vec3 cc_refl = get_prefiltered_color(worldspace + direction * 0.3, R, cc_roughness, NdotV, outirrad);
		vec4 cc_ssr = texture(ssr_accum, uv);
		cc_refl = mix(cc_refl, cc_ssr.rgb, clamp(cc_ssr.a, 0.0, 1.0));
		vec2 cc_brdf = texture(PBR_brdflut, vec2(NdotV, cc_roughness)).rg;
		spec_color *= vec3(1.0) - F_cc * cc_intensity;
		spec_color += cc_refl * (F_cc * cc_brdf.x + cc_brdf.y) * cc_intensity * spec_occlusion;
	}

	FS_OUT_Color = vec4(rad + spec_color, 1);
}
