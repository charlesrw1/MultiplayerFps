#ifndef SHADER_BUFFER_SHARED_
#define SHADER_BUFFER_SHARED_

// Per-pass UBO struct definitions shared between C++ and GLSL.
// Companion file to SharedGpuTypes.txt — same #ifdef __cplusplus pattern.
// See [[rendering/gfx_abstraction#2a]] for the Phase 2a UBO group plan.
//
// Convention:
//   - Struct name:   <Group>Params           (PascalCase)
//   - Shader block:  <Group>ParamsUbo { <Group>Params params; }
//   - C++ access:    gpu::<Group>Params p{};
//   - All groups share binding slot 7 (the per-pass push-constant analog).
//
// std140 padding rules (relevant subset):
//   - vec2 -> 8B align, vec3/vec4 -> 16B align, mat4 -> 16B align (4 vec4 rows)
//   - scalar (float/int/uint) -> 4B align
//   - struct -> 16B align, size padded to 16B multiple
//   - For the field shapes here (scalars / vec2 / vec3-with-pad / vec4 / mat4),
//     std140 and std430 produce identical layouts.

#ifdef __cplusplus
using namespace glm;
typedef uint32_t uint;
namespace gpu {
#endif

// ---- B. Bloom (pilot — done) -----------------------------------------------
// Used by BloomDownsampleF (srcResolution, mipLevel) and BloomUpsampleF
// (filterRadius). Both shaders iterate per-mip; each pass re-uploads.
struct BloomParams {
    vec2  srcResolution;
    int   mipLevel;
    float filterRadius;
};

// ---- G. Volumetric fog -----------------------------------------------------
// Used by VfogScatteringC (scatter pass) and VfogRaymarchC (raymarch pass).
// Bound at slot 4 (compute-shader UBO pool — distinct from the raster-side
// heightfog `FogUniforms` also at slot 4). The two shaders previously had
// per-file Vfog_Params copies with drifted field names (`density_ani` vs
// `den_ani`, `last_viewproj` vs `last_view`) — canonicalised here.
struct VfogParams {
    ivec4 volumesize;            //   0
    vec4  spread_frustumend;     //  16
    vec4  reprojection;          //  32
    mat4  last_viewproj;         //  48 .. 112
    vec4  density_ani;           // 112 .. 128
    int   num_lights;            // 128
    int   _pad0;                 // 132
    int   _pad1;                 // 136
    int   _pad2;                 // 140 .. 144 (struct size 144, /16)
};

// ---- F. Cull pipeline ------------------------------------------------------
// Union of fields set by name on the GPU cull / MDI compaction / depth-pyramid
// / debug-cull shaders. Each shader reads only its subset:
//   CullCompute       : lod_bias, force_lod, occlusion_cull, second_pass,
//                       radius_bias
//   compact_mdi       : num_draws, command_count
//   zero_instances_mdi: draw_count
//   debugCull         : lod_bias, output_depth, debug_pos
//   DepthPyramidC     : pyr_width, pyr_height, pyr_level (shared by cull + SSR hi-z)
// Cull dispatches 1-3x/frame, DepthPyramidC iterates per mip — each pass
// re-uploads. Bound at the per-pass slot 7.
// (Distinct from the existing CullDataBuffer at slot 5 which carries the
//  per-cull SSBO data: frustum planes, view matrix, pyramid dims.)
struct CullParams {
    vec4  debug_pos;          //   0 .. 16   (debugCull)
    int   lod_bias;           //  16         (CullCompute + debugCull)
    int   force_lod;          //  20         (CullCompute)
    int   occlusion_cull;     //  24         (CullCompute; bool, 4B std140)
    int   second_pass;        //  28         (CullCompute; bool)
    float radius_bias;        //  32         (CullCompute)
    int   output_depth;       //  36         (debugCull; bool)
    int   num_draws;          //  40         (compact_mdi)
    int   command_count;      //  44         (compact_mdi)
    int   draw_count;         //  48         (zero_instances_mdi)
    int   pyr_level;          //  52         (DepthPyramidC)
    float pyr_width;          //  56         (DepthPyramidC)
    float pyr_height;         //  60         (DepthPyramidC)
    int   cmd_offset;         //  64 .. 68   (zero_instances_mdi: draw_cmd_out[cmd_offset + gID]; struct size 64, /16)
};

// ---- D. SSR pipeline -------------------------------------------------------
// Union of fields previously set by name on ssr_f (the main HiZ/raymarch
// pass), ssr_downsample (color-mipchain per-mip), and ssr_upsample (full-res
// reconstruction). Each shader reads only the fields it needs:
//   ssr_f          : lastViewProj, g_proj, MAX_STEPS, max_distance, bias,
//                    step_size, max_thickness, temporalTime,
//                    ssr_max_roughness, ssr_brdf_bias, ssr_mip_bias,
//                    random_repeat, max_ssr_iters, num_traces_per_pixel,
//                    debug_toggle, res_mode, texel_offset
//   ssr_downsample : mip_level, myimg_size, inv_prev_size
//   ssr_upsample   : temporal_frame, num_samples_to_get, debug_toggle, res_mode
// ssr_downsample re-uploads per-mip in the loop; the other two upload once
// per frame. Bound at the per-pass slot 7.
// (reflectionShared.txt also declares uniforms but is shared with groups
//  A and E — its lum_adjust_mode / num_cubemaps don't live in SsrParams.)
struct SsrParams {
    mat4  lastViewProj;          //   0 .. 64    (ssr_f)
    mat4  g_proj;                //  64 .. 128   (ssr_f)
    int   MAX_STEPS;             // 128
    float max_distance;          // 132
    float bias;                  // 136
    float step_size;             // 140
    float max_thickness;         // 144
    float temporalTime;          // 148
    float ssr_max_roughness;     // 152
    float ssr_brdf_bias;         // 156
    float ssr_mip_bias;          // 160
    int   random_repeat;         // 164
    int   max_ssr_iters;         // 168         (ssr_f; default 100)
    int   num_traces_per_pixel;  // 172         (ssr_f; default 1)
    int   debug_toggle;          // 176         (ssr_f + ssr_upsample; bool)
    int   res_mode;              // 180         (ssr_f + ssr_upsample)
    int   temporal_frame;        // 184         (ssr_upsample)
    int   num_samples_to_get;    // 188         (ssr_upsample)
    int   mip_level;             // 192         (ssr_downsample)
    int   _pad0;                 // 196         pad before ivec2 (8B align)
    ivec2 texel_offset;          // 200 .. 208  (ssr_f)
    vec2  myimg_size;            // 208 .. 216  (ssr_downsample)
    vec2  inv_prev_size;         // 216 .. 224  (ssr_downsample; struct size 224, /16)
};

// ---- J. Probe / GI baking --------------------------------------------------
// Bake-time (editor) compute shaders. Each shader reads only its subset:
//   trace_C (incl. RELOCATE variant) : ray_sample_randomness, do_irrad_calcs,
//                                       num_lights, light_sun_dir, light_sun_color
//   gather_C                          : num_runs_so_far
//   probeLumCalc_C                    : num_cubemaps  (this is the per-bake
//                                       count, *not* the per-frame one set
//                                       by set_reflection_uniforms())
// One UBO; each dispatch re-uploads. Bound at slot 7.
// avgProbeCalc_C lives in the same dispatch family but has no name-uniforms,
// so it doesn't need a slot in this struct.
struct BakeParams {
    vec4  light_sun_dir;          //   0 .. 16  (trace_C)
    vec4  light_sun_color;        //  16 .. 32  (trace_C)
    float ray_sample_randomness;  //  32        (trace_C + RELOCATE)
    int   do_irrad_calcs;         //  36        (trace_C; bool, 4B std140)
    int   num_lights;             //  40        (trace_C)
    int   num_runs_so_far;        //  44        (gather_C)
    int   num_cubemaps;           //  48        (probeLumCalc_C)
    int   _pad0;                  //  52
    int   _pad1;                  //  56
    int   _pad2;                  //  60 .. 64  (struct size 64, /16)
};

// ---- A. Lit + compositor ---------------------------------------------------
// Active lit/compositor shaders (sun, ambient, tile-lighting, tonemap composite).
// Each shader reads only its subset:
//   SunLightAccumulationF       : uSunDirection, uSunColor, uEpsilon
//   AmbientLightingF            : sky_color, ground_color
//   LightAccumulationFullScreen : num_lights, light_indirect_offset
//                                 (TILED=1 variant reads its own locals from
//                                  the tiled SSBO; the per-tile loop in
//                                  LightListCuller::draw_lights re-uploads
//                                  for TILED=2)
//   CombineF                    : tonemap_type, contrast_tweak, saturation_tweak,
//                                 bloom_lerp, exposure
// Note: the shared `specular_ao_intensity` from reflectionShared.txt lives
// in DdgiRuntimeParams — reflectionShared is included only by group E shaders.
// Bound at slot 7. TILED=2 loop re-uploads per tile.
struct LitCompositorParams {
    vec4 uSunDirection;          //   0 .. 16  (xyz used; SunLightAccumulationF)
    vec4 uSunColor;              //  16 .. 32  (xyz used; SunLightAccumulationF)
    vec4 sky_color;              //  32 .. 48  (xyz used; AmbientLightingF)
    vec4 ground_color;           //  48 .. 64  (xyz used; AmbientLightingF)
    float uEpsilon;              //  64        (SunLightAccumulationF)
    float contrast_tweak;        //  68        (CombineF)
    float saturation_tweak;      //  72        (CombineF)
    float bloom_lerp;            //  76        (CombineF)
    float exposure;              //  80        (CombineF)
    int   tonemap_type;          //  84        (CombineF)
    int   num_lights;            //  88        (LightAccumulationFullScreen TILED=0/2)
    int   light_indirect_offset; //  92 .. 96  (LightAccumulationFullScreen TILED=2; size 96, /16)
};

// ---- E. DDGI runtime -------------------------------------------------------
// Active DDGI lighting/composite pass shaders.
//   ddgiShadeF (regular + HALFRES_DDGI variants) : selected_probe_pos,
//     irrad_mult, include_cubemaps, halfres_texel_offset (HALFRES only),
//     + reflectionShared fields (specular_ao_intensity, lum_adjust_mode,
//     num_cubemaps).
//   ddgi_apply_upsampled : specular_ao_intensity, ssr_lum_range +
//     reflectionShared fields.
//   reflectionShared.txt is a helper include — it relies on the including
//   shader having bound DdgiRuntimeParams at slot 7, and reads
//   params.lum_adjust_mode / params.num_cubemaps directly.
// Bound at slot 7. Each callsite uploads its full subset before dispatch.
struct DdgiRuntimeParams {
    vec2  ssr_lum_range;         //   0 .. 8   (ddgi_apply_upsampled)
    int   _pad0;                 //   8        pad to ivec4 16B align
    int   _pad1;                 //  12
    ivec4 selected_probe_pos;    //  16 .. 32  (ddgiShadeF; xyz used)
    float specular_ao_intensity; //  32        (ddgiShadeF + ddgi_apply_upsampled + reflectionShared)
    float irrad_mult;            //  36        (ddgiShadeF)
    int   lum_adjust_mode;       //  40        (reflectionShared)
    int   num_cubemaps;          //  44        (reflectionShared)
    int   include_cubemaps;      //  48        (ddgiShadeF; bool, 4B std140)
    int   _pad2;                 //  52        pad before ivec2 (8B align)
    ivec2 halfres_texel_offset;  //  56 .. 64  (ddgiShadeF HALFRES; struct size 64, /16)
};

// ---- H. SSAO / HBAO pipeline -----------------------------------------------
// HBAO from NVIDIA gl_ssao sample. The HBAOData SSBO at slot 0 holds the
// big per-frame constants (radii, jitters, projInfo derivatives, etc.);
// this struct is the small per-pass tweakables that were previously set
// by name on each individual sub-pass:
//   linearizedepth.txt  : clipInfo, zNear
//   viewnormal.txt      : projInfo, projOrtho, InvFullResolution
//   hbaodeinterleave.txt: info
//   hbao.txt            : primitive_id_custom (per-layer in calc loop)
//   hbaoblur.txt        : g_Sharpness, g_InvResolutionDirection (per-direction)
// Bound at slot 7. Multiple uploads per frame: once each for the linearize +
// viewnormal + deinterleave + blurH + blurV passes, and once per array
// layer inside the calc loop.
struct SsaoHbaoParams {
    vec4  clipInfo;                  //   0 .. 16  (linearizedepth)
    vec4  projInfo;                  //  16 .. 32  (viewnormal)
    vec4  info;                      //  32 .. 48  (hbaodeinterleave; xy offset, zw invRes)
    vec2  InvFullResolution;         //  48 .. 56  (viewnormal)
    vec2  g_InvResolutionDirection;  //  56 .. 64  (hbaoblur)
    float zNear;                     //  64        (linearizedepth)
    float g_Sharpness;               //  68        (hbaoblur)
    int   projOrtho;                 //  72        (viewnormal)
    uint  primitive_id_custom;       //  76 .. 80  (hbao; struct size 80, /16)
};

// ---- C. Temporal upsample --------------------------------------------------
// Union of fields previously set by name on TaaResolveF, temporal_upsample_ddgi,
// and temporal_upsample_ssr. Each shader reads only the fields it needs:
//   TAA   : lastViewProj, amt, doc_*, remove_flicker, use_reproject, dilate_velocity
//   DDGI  : same + halfres_texel_offset
//   SSR   : same + halfres_texel_offset + ssr_nonoccluded_weight_mult
// One UBO storage, three call sites — three uploads/frame, all under slot 7.
struct TemporalParams {
    mat4 lastViewProj;                  //   0 .. 64
    float amt;                          //  64
    float doc_mult;                     //  68
    float doc_vel_bias;                 //  72
    float doc_bias;                     //  76
    float doc_pow;                      //  80
    float ssr_nonoccluded_weight_mult;  //  84
    int   remove_flicker;               //  88   bool (4B in std140)
    int   use_reproject;                //  92
    int   dilate_velocity;              //  96
    int   _pad0;                        // 100   pad before ivec2 (8B align)
    ivec2 halfres_texel_offset;         // 104 .. 112  (struct size 112, /16)
};

// ---------------------------------------------------------------------------
// Push-constant blocks (Phase 2 B1).
// ---------------------------------------------------------------------------
// Per-shader, per-draw uniform data uploaded via
// gfx().push_{vertex,fragment,compute}_constants(slot, &pc, sizeof(pc)).
// Mirrors SDL3 GPU's SDL_PushGPU*UniformData. Max 128 B per block.
//
// Binding mapping (see IGraphicsDevice.h::kGfxPushConstBindingBase):
//   VS slot 0..3 -> bindings 12..15
//   FS slot 0..3 -> bindings 16..19
//   CS slot 0..3 -> bindings 20..23
//
// GLSL declaration template (one block per stage that uses push consts):
//   layout(std140, binding = 12) uniform XVertPushConstsUbo { XVertPushConsts pcv; };
//   ... vert shader reads pcv.ViewProj, pcv.Model, ...
//
// Block instance name convention: ALWAYS stage-specific -- `pcv` (vertex),
// `pcf` (fragment), `pcc` (compute). UBO block instance names are program-
// global to the GL linker, so identical instance names in different stages
// of the same program collide ("Anonymous member name used for global
// variable or other anonymous member").

// MbSimpleV.txt / MbSimpleF.txt (debug meshbuilder line draws -- see
// DrawLocal_Lighting.cpp). Vert needs Model + ViewProj; frag needs an
// optional solid override colour (USE_SOLID_COLOR define).
struct MbSimpleVertPushConsts {
    mat4 ViewProj;               //   0 ..  64
    mat4 Model;                  //  64 .. 128   (struct size 128)
};
struct MbSimpleFragPushConsts {
    vec4 solid_color;            //   0 ..  16   (struct size 16)
};

// MeshSimpleV.txt + MeshDebugProbeF.txt (DDGI probe debug draws -- see
// RaytraceTest_Shade.cpp). VS needs per-probe Model; FS needs the probe
// grid coordinate, full grid dims, and a global offset into the irradiance
// atlas. ivec3 fields wrapped to ivec4 to keep std140 layout obvious.
struct MeshSimpleVertPushConsts {
    mat4 Model;                  //   0 ..  64   (struct size 64)
};
struct MeshDebugProbeFragPushConsts {
    ivec4 probe_coord;           //   0 ..  16   (.xyz used)
    ivec4 vol_grid;              //  16 ..  32   (.xyz used)
    int   vol_offset;            //  32 ..  36
    int   _pad0;
    int   _pad1;
    int   _pad2;                 //   (struct size 48)
};

// MasterUIShader.txt (UI material draws -- see DrawLocal_Misc.cpp).
struct MasterUIVertPushConsts {
    mat4 UIViewProj;             //   0 ..  64   (struct size 64)
};

// MasterDecalShader.txt (decal MDI batches -- see DecalBatcher.cpp).
struct MasterDecalVertPushConsts {
    uint decal_indirect_offset;  //   0 ..   4   (struct size 16, padded)
    uint _pad0;
    uint _pad1;
    uint _pad2;
};

// fullscreen_quad_textureF.txt (light-cookie atlas blits -- see
// RenderExtra_LightCookieAtlas.cpp::blit_texture_into_thing_because_reasons).
struct FullscreenDrawTextureFragPushConsts {
    ivec2 viewport_size;         //   0 ..   8
    int _pad0;
    int _pad1;                   //   (struct size 16)
};

// Helpers/EqrtCubemapV.txt (env-probe cubemap face renders) +
// Helpers/PrefilterSpecularNewF.txt (specular prefilter) -- see EnvProbe.cpp.
struct EqrtCubemapVertPushConsts {
    mat4 ViewProj;               //   0 ..  64   (struct size 64)
};
struct PrefilterSpecularFragPushConsts {
    float roughness;             //   0 ..   4
    float _pad0;
    float _pad1;
    float _pad2;                 //   (struct size 16)
};

// MasterDeferredShader.txt (every opaque MDI batch in setup_batch -- see
// DrawLocal_RenderPass.cpp). VS reads indirect_material_offset to index
// indirect_materials[]; FS reads the same value only under
// DEBUG_SHADER + DEBUG_DRAWID for draw-id colour visualisation. Single
// struct mirrored across both stages (push_vertex_constants and
// push_fragment_constants get the same blob per draw).
struct MasterDeferredPushConsts {
    uint indirect_material_offset; //   0 ..   4   (struct size 16, padded)
    uint _pad0;
    uint _pad1;
    uint _pad2;
};

// MbTexturedV.txt / MbTexturedF.txt (debug texture viewer -- see
// Renderer::draw_output_tex in DrawLocal_Debug.cpp). VS needs Model +
// ViewProj; FS needs alpha + mip_slice (other named uniforms in the file
// stay at compile-time defaults and have no C++ callers).
struct MbTexturedVertPushConsts {
    mat4 ViewProj;               //   0 ..  64
    mat4 Model;                  //  64 .. 128   (struct size 128)
};
struct MbTexturedFragPushConsts {
    float alpha;                 //   0 ..   4
    float mip_slice;             //   4 ..   8
    float _pad0;
    float _pad1;                 //   (struct size 16)
};

// MasterParticleShader.txt (render_particles -- see DrawLocal_RenderPass.cpp).
// VS needs Model + ViewProj; FS exposes FS_IN_Matid to user fragment code.
struct MasterParticleVertPushConsts {
    mat4 ViewProj;               //   0 ..  64
    mat4 Model;                  //  64 .. 128   (struct size 128)
};
struct MasterParticleFragPushConsts {
    uint FS_IN_Matid;            //   0 ..   4   (struct size 16, padded)
    uint _pad0;
    uint _pad1;
    uint _pad2;
};
