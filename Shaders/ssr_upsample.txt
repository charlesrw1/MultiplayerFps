float GGX_D(float NoH, float alpha)
{
    float alpha2 = alpha * alpha;
    float denom = NoH*NoH*(alpha2 - 1.0) + 1.0;
    return alpha2 / (3.14159265 * denom * denom);
}

// PDF of GGX half-vector H
float GGX_PDF(vec3 H, vec3 N, vec3 V, float roughness)
{
    float alpha = roughness * roughness;
    float NoH = max(dot(N, H), 0.0);
    float VoH = max(dot(V, H), 0.0);
    
    float D = GGX_D(NoH, alpha);
    
    // convert half-vector PDF -> reflection vector PDF
    return D * NoH / (4.0 * VoH + 1e-5);
}

vec3 get_position_from_depth(ivec2 sample_texel, float depth)
{
	const vec2 uv = (vec2(sample_texel))/vec2(g.viewport_size.xy);
	const vec4 clippos = vec4(uv * 2.0 - 1.0, depth, 1.0);
    const vec4 viewPos4 = g.invproj * clippos;
    const vec3 viewPos = viewPos4.xyz / viewPos4.w;
	return viewPos.xyz;
}
vec3 get_position_from_depth_uv(vec2 sample_uv, float depth)
{
	const vec2 uv = sample_uv;
	const vec4 clippos = vec4(uv * 2.0 - 1.0, depth, 1.0);
    const vec4 viewPos4 = g.invproj * clippos;
    const vec3 viewPos = viewPos4.xyz / viewPos4.w;
	return viewPos.xyz;
}

vec3 pbr_calculate_reflection2(vec3 Albedo, float Roughness, float Metalness, vec3 Normal, vec3 V, vec3 prefiltered)
{
        vec3 R = reflect(-V,Normal);
vec3 F0 = vec3(0.04);
        F0 = mix(F0,Albedo,Metalness);

        //vec3 prefiltered_color = textureLod(skyprobe,vec3(R), Roughness*MAX_REFLECTION_LOD).rgb;
        //vec3 prefiltered_color = pow(ref_irrad,vec3(1.5));
        float NdotV = max(dot(Normal,V),0.0);

        vec3 prefiltered_color = prefiltered;// textureLod(probes,vec4(R,float(probe_num)),Roughness*MAX_REFLECTION_LOD).rgb;
        //prefiltered_color *= ref_irrad;



        vec3 F = FresnelSchlickRougness(NdotV,F0,Roughness);


        //float F90 = clamp(50*dot(F0,vec3(0.33)),0,1);
        //vec3 F = modified_f_schlick(F0,F90,NdotV);

        vec2 env_brdf = texture(pbrLUT,vec2(max(dot(Normal,V),0.0),Roughness)).rg;


        vec3 specular = prefiltered_color*(F*env_brdf.x+env_brdf.y);
     
        return specular;
}

vec3 SoftClamp(vec3 color, float threshold)
{
    vec3 over = max(color - threshold, vec3(0.0));
    vec3 compressed = threshold + over / (1.0 + over); // compress high values
    return min(color, compressed); 
}

		const vec3 uv_and_pdf = texelFetch(hitbuffer,sample_texel, 0).rgb;
		if(uv_and_pdf.z<=1e-6)
			continue;
		count += 1;
			
		//const float sample_rough = texelFetch(GBUFFER2,sample_texel,0).g;

		//const ivec2 sample_hit_texel = ivec2(uv_and_pdf.xy * g.viewport_size.xy);
		const vec3 sample_hit_vs = get_position_from_depth_uv(uv_and_pdf.xy,
			textureLod(DEPTH,uv_and_pdf.xy,0).r);
			
		
		const vec3 sample_r_vec_vs = normalize(sample_hit_vs-MY_viewPos);

	}
	
	float preFactor = texture(pbrLUT,vec2(max(dot(MYnormalVS,-MY_viewDir),0.0),MY_roughness)).r;
	float normalizedWeight = min(sum_weight * preFactor, 1.0);
	if(sum_weight>=1e-5) {

		sum *= sum_weight * preFactor;
	}
		
		

	
	vec3 blah = texelFetch(hitbuffer,texel,0).rgb;
