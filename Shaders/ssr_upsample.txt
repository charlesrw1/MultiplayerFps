
layout(location = 0) in vec2 texCoord;
layout(location = 0) out vec4 FS_OUT_Color;

#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"
layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};
layout (binding = 7, std140) uniform SsrParamsUbo { SsrParams params; };

layout(binding=0) uniform sampler2D traceBuffer;

float Luminance(vec3 c) { return dot(c, vec3(0.2126, 0.7152, 0.0722)); }

vec2 RandN2(vec2 n)
{
	return fract(sin(dot(n, vec2(12.9898, 78.233))) * vec2(43758.5453123, 28001.8384));
}

void main()
{
	vec2 uv = texCoord;

	const vec2 offsets[8] = vec2[8](
		vec2( 0,  0),
		vec2( 1, -1),
		vec2(-1, -1),
		vec2( 0,  1),
		vec2(-1,  0),
		vec2( 0, -1),
		vec2( 1,  0),
		vec2( 1,  1)
	);

	// Randomize offset rotation
	vec2 random = RandN2(uv + params.temporalTime);
	vec2 blueNoise = random * 2.0 - 1.0;
	mat2 offsetRotationMatrix = mat2(blueNoise.x, blueNoise.y, -blueNoise.y, blueNoise.x);

	vec4 result = vec4(0.0);
	int sampleCount = params.resolveSamples;
	for (int i = 0; i < sampleCount; i++)
	{
		vec2 offsetUV = offsets[i] * params.ssrTexelSize;
		offsetUV = offsetRotationMatrix * offsetUV;
		vec4 value = textureLod(traceBuffer, uv + offsetUV, 0.0);
		// Luminance weight to reduce fireflies
		value.rgb /= 1.0 + Luminance(value.rgb);
		result += value;
	}

	result /= float(sampleCount);
	result.a *= params.intensity;
	// Undo luminance weight
	result.rgb /= 1.0 - Luminance(result.rgb);

	// Normalize alpha
	result.rgb *= max(1.0, result.a);
	result.a = min(result.a, 1.0);

	FS_OUT_Color = result;
}
