// for sampling the sky



// each thread = 1 probe (ie global id)
// each thread computes 64 samples, computes lighting. then saves to work buffer

// then run again to save into atlas

// repeat.

// need: radiance, depth, hitbuffer, lighting (lights, sun, spot, skybox), material/raytrace stuff

// volume. (box, transform, probe dim)



#define DDGILIGHTING
#include "ddgiShared.txt"


#include "pbrShared.txt"

float luminance(vec3 v)
{
    return dot(v, vec3(0.2126, 0.7152, 0.0722));
}

void main()
{
	const int global_index = int(gl_GlobalInvocationID.x);
	if(global_index>=num_cubemaps)
		return;
	
	vec3 boxmin = Cubemaps.data[global_index].bounds_min.rgb;
	vec3 boxmax = Cubemaps.data[global_index].bounds_max.rgb;
	vec3 middle = (boxmax+boxmin)*0.5;
	
	vec3 sum = vec3(0.0);
	
	sum += calculate_irradiance(vec3(1,0,0),middle,middle);
	sum += calculate_irradiance(vec3(-1,0,0),middle,middle);
	sum += calculate_irradiance(vec3(0,1,0),middle,middle);
	sum += calculate_irradiance(vec3(0,-1,0),middle,middle);
	sum += calculate_irradiance(vec3(0,0,1),middle,middle);
	sum += calculate_irradiance(vec3(0,0,-1),middle,middle);
	
	sum /= 6.0;
	
	float lum = luminance(sum);
	
	
	Cubemaps.data[global_index].bounds_min = vec4(boxmin,lum);
}
