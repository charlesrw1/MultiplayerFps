#include "ShaderBufferShared.txt"

layout (location = 0) in vec3 VS_IN_Postion;
layout (location = 1) in vec4 VS_IN_Color;
layout (location = 2) in vec2 VS_IN_TexCoord;

layout(location = 0) out vec4 FS_IN_Color;
layout(location = 1) out vec3 FS_IN_Fragpos;

layout(std140, binding = 12) uniform MbSimpleVertPushConstsUbo { MbSimpleVertPushConsts pcv; };

void main()
{
	FS_IN_Color = VS_IN_Color;
	FS_IN_Fragpos = VS_IN_Postion;
	vec4 pos =  pcv.ViewProj *  pcv.Model * vec4(VS_IN_Postion, 1.0);

	#ifdef SKYBOX
	gl_Position = pos.xyww;	// so z will always = 1
	#else
	gl_Position = pos;
	#endif
}
