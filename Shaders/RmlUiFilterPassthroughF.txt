#include "ShaderBufferShared.txt"

layout(location = 0) in vec2 texCoord;
layout(location = 0) out vec4 FS_OUT_Color;

layout(binding = 0) uniform sampler2D the_texture;

layout(std140, binding = 16) uniform RmlUiFilterPassthroughFragPushConstsUbo { RmlUiFilterPassthroughFragPushConsts pcf; };

void main()
{
	FS_OUT_Color = texture(the_texture, texCoord) * pcf.opacity;
}
