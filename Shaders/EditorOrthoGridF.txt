layout(location = 0) out vec4 FS_OUT_Color;
layout(location = 0) in vec2 texCoord;

layout(binding = 0) uniform sampler2D DEPTH;

#include "SharedGpuTypes.txt"

layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};

#include "commonUtil.txt"

float grid_line(float coord, float line_width, float fw) {
	float d = abs(fract(coord + 0.5) - 0.5);
	return 1.0 - smoothstep(0.0, fw * line_width, d);
}

void main()
{
	float depth = texture(DEPTH, texCoord).r;
	if (depth > 0.0001)
		discard;

	vec3 worldPos = WorldPosFromDepth(texCoord, depth);

	vec3 viewDir = normalize(g.viewfront.xyz);
	vec3 absDir = abs(viewDir);

	vec2 gridUV;
	if (absDir.y > absDir.x && absDir.y > absDir.z)
		gridUV = worldPos.xz;
	else if (absDir.x > absDir.z)
		gridUV = worldPos.zy;
	else
		gridUV = worldPos.xy;

	float snap = g.editor_grid_snap;
	if (snap < 0.001) snap = 1.0;

	// ortho width from inverse projection
	float ortho_width = abs(g.invproj[0][0]) * 2.0;
	float pixels_per_unit = g.viewport_size.x / ortho_width;

	// minor spacing = snap, major = snap * 10
	// as you zoom out, promote: minor becomes snap*10, major becomes snap*100, etc.
	// pick the level where minor lines are ~40-80px apart on screen
	float minor_px = snap * pixels_per_unit;
	float level = 0.0;
	while (minor_px < 4.0 && level < 10.0) {
		minor_px *= 10.0;
		level += 1.0;
	}
	float minor_spacing = snap * pow(10.0, level);
	float major_spacing = minor_spacing * 10.0;

	// fade: how close minor lines are to being too dense (for smooth transition)
	float minor_screen_px = minor_spacing * pixels_per_unit;
	float fade = clamp((minor_screen_px - 4.0) / 40.0, 0.0, 1.0);

	vec2 fw = fwidth(gridUV);

	// minor grid lines
	float minor_x = grid_line(gridUV.x / minor_spacing, 1.2, fw.x / minor_spacing);
	float minor_y = grid_line(gridUV.y / minor_spacing, 1.2, fw.y / minor_spacing);
	float minor = max(minor_x, minor_y);

	// major grid lines
	float major_x = grid_line(gridUV.x / major_spacing, 1.8, fw.x / major_spacing);
	float major_y = grid_line(gridUV.y / major_spacing, 1.8, fw.y / major_spacing);
	float major = max(major_x, major_y);

	float minor_alpha = minor * fade * 0.7;
	float major_alpha = major * 0.8;

	vec3 bg = vec3(0.12);
	vec3 minor_color = vec3(0.6);
	vec3 major_color = vec3(0.6);

	vec3 color = bg;
	color = mix(color, minor_color, minor_alpha);
	color = mix(color, major_color, major_alpha);

	FS_OUT_Color = vec4(color, 1.0);
}
