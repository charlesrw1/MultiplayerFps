layout(location = 0) out vec4 FS_OUT_Color;

layout(binding = 0) uniform sampler2D GBUFFER0;		// ws normal (xyz)
layout(binding = 1) uniform sampler2D GBUFFER1;		// albedo, ao
layout(binding = 2) uniform sampler2D GBUFFER2;		// metallic,roughness, custom, matid
layout(binding = 3) uniform sampler2D DEPTH;		// depth texture (writes disabled)
layout(binding = 4) uniform sampler2DShadow shadow_map_atlas;
layout(binding = 5) uniform sampler2D light_cookie;


#include "SharedGpuTypes.txt"
#include "LightingShared.txt"

layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};

layout (binding = 1, std430) readonly buffer LightingObjectUniforms_Buffer {
	LightingObjectUniforms light_objects[];
};

layout (binding = 2, std430) readonly buffer LightTiledUniforms_Buffer {
	TiledLightUniforms tiled_data;
};
layout (binding = 3, std430) readonly buffer LightTiledCount_Buffer {
	int light_tile_counts[];
};
layout (binding = 4, std430) readonly buffer LightTiledIndicies_Buffer {
	int light_tile_indicies[];
};




#include "commonUtil.txt"
#include "pbrShared.txt"
#include "pointLightShared.txt"


#ifndef TILED
#define TILED 0
#endif

#if TILED == 0
uniform int num_lights = 0;
#elif TILED == 1

#elif TILED == 2
uniform int num_lights = 0;
uniform int light_indirect_offset = 0;
#endif

void main()
{
	vec2 uv = vec2(gl_FragCoord.xy) / vec2(g.viewport_size.xy);
	
	#if TILED == 1
		ivec2 tile_size = ivec2(g.viewport_size.xy)/ivec2(tiled_data.tile_count_x,tiled_data.tile_count_y);
		ivec2 this_tile = ivec2(vec2(gl_FragCoord.xy) * vec2(tiled_data.inv_tile_size_x,tiled_data.inv_tile_size_y));
		this_tile.y =tiled_data.tile_count_y - this_tile.y - 1;
		
		int tile_index = this_tile.y * tiled_data.tile_count_x + this_tile.x;
		int num_lights = light_tile_counts[tile_index];
		int light_indirect_offset = tile_index*MAX_TILE_LIGHTS;
		
	#endif
	
	//
	vec4 metal_rough = texture(GBUFFER2,uv);
	vec4 albedo = texture(GBUFFER1,uv);
	vec4 depth_tex = texture(DEPTH,uv);
	vec4 normal_t = texture(GBUFFER0,uv);
	
	uint type = get_gbuffer_material_type(metal_rough);
	if(type==MATERIAL_TYPE_UNLIT)
		discard;
		
	float depth_value = depth_tex.r;
	vec3 worldspace = WorldPosFromDepth(uv, depth_value);
	vec3 output_light_accum = vec3(0.0);
	for(int i=0;i<num_lights;i++) {
	#if TILED == 1
		int light_index = light_tile_indicies[light_indirect_offset+i];
	#elif TILED == 2
		int light_index = light_tile_indicies[light_indirect_offset+i];
	#else
		int light_index = i;
	#endif
	
		vec4 _position_radius = light_objects[light_index].position_radius;
		vec3 position = _position_radius.xyz;
		float radius = _position_radius.w;
		vec3 color = light_objects[light_index].light_color.xyz;
		bool is_spot_light = bool(light_objects[light_index].flags & 1);
		bool casts_shadow = bool(light_objects[light_index].flags & 2);
		vec3 spot_normal =  light_objects[light_index].spot_normal.xyz;
		float spot_angle = light_objects[light_index].spot_angle;
		float spot_inner = light_objects[light_index].spot_inner;


		
		float distance_to_light = length(position-worldspace);
		if(distance_to_light>radius)
			continue;
		
		vec3 L = (position - worldspace)/distance_to_light;
		
		float distance_to_light_2 = distance_to_light*distance_to_light;
		float radius_2 = radius*radius;
		float attenuation =punctual_attenuation(distance_to_light_2,radius_2);
		
		vec3 out_color = pbr_calculate_punctual(color, L, worldspace, albedo.xyz, metal_rough.y, metal_rough.x, normal_t.xyz);
		
		if(is_spot_light) {
			float angle = dot(-L, spot_normal);
		#ifndef COOKIE
			attenuation *= spotlight_angle_atten(spot_angle,angle,spot_inner);//clamp((-spot_angle+angle) / (spot_inner-spot_angle), 0.0, 1.0);    
		#endif
		#ifdef SHADOWED
			if(casts_shadow) {
				attenuation *= sample_shadow(worldspace, light_index);
			}
		#endif
			
		}
		
		output_light_accum += out_color * attenuation;
	}
	
	
	FS_OUT_Color = vec4(output_light_accum,1.0);//vec4(linearize_depth(depth_value,0.01,100.0)*0.02);
}
#endif
