float ComputeScattering(float lightDotView)
{
	float scattering = 0.001;
	float result=1.0-scattering*scattering;
	result /= (4.0*PI*pow(1.0+scattering*scattering-(2.0*scattering)*lightDotView,1.5));
	return result;
}

float phase_function(vec3 Wo, vec3 Wi, float g)
{
    float cos_theta = dot(Wo, Wi);
    float denom     = 1.0 + g * g + 2.0 * g * cos_theta;
    return (1.0 / (4.0 * PI)) * (1.0 - g * g) / max(pow(denom, 1.5), 0.0001);
}

vec3 get_worldpos_new(ivec3 fragcoords,vec3 invpixelsize, vec3 poffset)
{
	vec3 screencoord = (vec3(fragcoords)+poffset)*invpixelsize;	// [0,1]
	
	float tilez = get_world_depth_from_volume_normalized(screencoord.z,fogparam.spread_frustumend.x,fogparam.spread_frustumend.y);
	float clipspace_depth = unlinearize_depth(tilez);	// projection space
	vec3 ndc = vec3(screencoord.x*2.0 - 1.0, screencoord.y*2.0 - 1.0, clipspace_depth);
	
	vec4 world = g.inv_viewproj * vec4(ndc,1.0);
	world/=world.w;

	return world.xyz;
}

vec3 calculate_irradiance2(vec3 direction, vec3 worldspace, vec3 biased_world_position)
{
	const int volume_index = find_best_ddgi_volume(worldspace);
	const vec3 volume_origin = volumes[volume_index].origin_pri.xyz;
	const vec3 volume_spacing = volumes[volume_index].density.xyz;
	const ivec3 vol_grid = volumes[volume_index].size_offset.xyz;
	const int vol_offset = volumes[volume_index].size_offset.w;
	

	ivec3 base_probe_coords = ivec3((biased_world_position-volume_origin)/volume_spacing);
	base_probe_coords = clamp(base_probe_coords,ivec3(0), 
	ivec3(vol_grid)-ivec3(1)
	);
	vec3 base_probe_pos = get_probe_world_from_coords(base_probe_coords, volume_origin,volume_spacing, vol_grid, vol_offset);
	
	vec3 alpha = clamp(
		(biased_world_position-base_probe_pos)/volume_spacing,
		vec3(0),
		vec3(1));
	

	vec3 irradiance=vec3(0.0);
	float accum_weight = 0.0;
	float cheb_weight_total = 0.0;
	
	for(int probeIndex = 0;probeIndex < 8; probeIndex++) {
		ivec3 adjacent_probe_ofs = ivec3(probeIndex, probeIndex>>1, probeIndex>>2)&ivec3(1);
		ivec3 adjacent_probe_coords = clamp(
			base_probe_coords+adjacent_probe_ofs,
			ivec3(0),
			ivec3(vol_grid)-ivec3(1));
		vec3 adj_probe_worldpos = get_probe_world_from_coords(adjacent_probe_coords, volume_origin,volume_spacing, vol_grid, vol_offset);
		

		vec3 biased_pos_to_adj_probe = normalize(adj_probe_worldpos-biased_world_position);
		float biased_pos_to_adj_probe_len = length(adj_probe_worldpos-biased_world_position);
		
		vec3 trilinear = max(vec3(0.001), 
			mix(vec3(1.0-alpha),vec3(alpha),vec3(adjacent_probe_ofs)));
		float trilinear_weight = trilinear.x*trilinear.y*trilinear.z;
		float weight = 1.0;
		float warp_shading = dot(biased_pos_to_adj_probe,direction)*0.5 + 0.5;
		weight *= (warp_shading*warp_shading)+0.2;
		
		vec2 octant_coords = OctahedralEncode(-biased_pos_to_adj_probe);
		
		vec2 probe_texture_uv = get_depth_probe_uv(
		octant_coords,
		adjacent_probe_coords, vol_grid, vol_offset);
		
		//vec2 filtered_distance = vec2(2.0)*texture(probe_depth, probe_texture_uv).rg;
		//
		//float variance = abs(
		//(filtered_distance.x*filtered_distance.x) - filtered_distance.y);
		//
		//float chebyshev_weight = 1.0;
		//if(biased_pos_to_adj_probe_len > filtered_distance.x) {
		//	float v = biased_pos_to_adj_probe_len - filtered_distance.x;
		//	chebyshev_weight = variance/(variance+(v*v));
		//	chebyshev_weight = max(
		//	chebyshev_weight*chebyshev_weight*chebyshev_weight,
		//	0.0
		//	);
		//}
		
		//weight *= max(0.05, chebyshev_weight);
		

		weight = max(0.000001, weight);
		const float crush_thresh = 0.2;
		if(weight<crush_thresh) {
			weight *= (weight*weight)*(1.0/(crush_thresh*crush_thresh));
		}
		weight *= trilinear_weight;
		
	
		const int probe_index = adjacent_probe_coords.x + adjacent_probe_coords.y*vol_grid.x + adjacent_probe_coords.z*vol_grid.x*vol_grid.y + vol_offset;
		
		vec3 probe_irrad=vec3(0.0);
		probe_irrad.r = ddgi_probe_avg_value[probe_index*3];
		probe_irrad.g = ddgi_probe_avg_value[probe_index*3+1];
		probe_irrad.b = ddgi_probe_avg_value[probe_index*3+2];
		
		
	
	
		
		irradiance += (weight*probe_irrad);
		accum_weight += weight;
		// holy crap so much code!
		
		//cheb_weight_total+= chebyshev_weight;
	}
	
	irradiance *= (1.0/accum_weight);
	
	return irradiance;

}
vec3 get_inscattering(vec3 worldspace)
{
	vec3 raydir = normalize(worldspace-g.viewpos_time.xyz);


	vec3 accum = vec3(0.0);
	for(int light_index = 0;light_index<num_lights;light_index++) {
	
		vec4 _position_radius = light_objects[light_index].position_radius;
		vec3 position = _position_radius.xyz;
		float radius = _position_radius.w;
		vec3 color = light_objects[light_index].light_color.xyz;
		bool is_spot_light = bool(light_objects[light_index].flags & 1);
		bool casts_shadow = bool(light_objects[light_index].flags & 2);
		vec3 spot_normal =  light_objects[light_index].spot_normal.xyz;
		float spot_angle = light_objects[light_index].spot_angle;
		float spot_inner = light_objects[light_index].spot_inner;
		
		float distance_to_light = length(position-worldspace);
		if(distance_to_light>radius)
			continue;
		
		vec3 L = (position - worldspace)/distance_to_light;
		
		distance_to_light = max(distance_to_light,0.2);
		
		float distance_to_light_2 = distance_to_light*distance_to_light;
		float radius_2 = radius*radius;
		float attenuation =punctual_attenuation(distance_to_light_2,radius_2);
				
		if(is_spot_light) {
			float angle = dot(-L, spot_normal);
		#ifndef COOKIE
			attenuation *= spotlight_angle_atten(spot_angle,angle,spot_inner);//clamp((-spot_angle+angle) / (spot_inner-spot_angle), 0.0, 1.0);    
		#endif
	
			if(casts_shadow) {
				attenuation *= sample_shadow(worldspace, light_index);
			}
			
		}

		accum += attenuation * color * phase_function(raydir, -L,fogparam.density_ani.y);
	
	}

	//vec3 spotdir = worldpos-(spot_pos+vec3(2,0,0));
	//float spotlen = length(spotdir);
	//spotdir = spotdir/spotlen;
	

	vec3 sum = vec3(0.0);
	vec3 biased_towards_cam = worldspace-raydir*0.4;
	sum += calculate_irradiance2(-raydir,biased_towards_cam,biased_towards_cam);
	//sum += calculate_irradiance(vec3(-1,0,0),worldspace,worldspace);
	//sum += calculate_irradiance(vec3(0,1,0),worldspace,worldspace);
	//sum += calculate_irradiance(vec3(0,-1,0),worldspace,worldspace);
	//sum += calculate_irradiance(vec3(0,0,1),worldspace,worldspace);
	//sum += calculate_irradiance(vec3(0,0,-1),worldspace,worldspace);	
	//sum /= 6.0;
	accum += sum;
	//if(dot(spotdir,spotlightnormal)>0.5) {
		//spotadd = phase_function(raydir,spotdir,anisotropy)/(spotlen*spotlen);
	//	spotadd = min(spotadd, 100.0);
	//}
	return accum;
}

void main()
{
	ivec3 fragcoords = ivec3(gl_GlobalInvocationID.xyz);
	vec3 invpixelsize = 1/vec3(fogparam.volumesize);

	vec3 worldpos = get_worldpos_new(fragcoords, invpixelsize, vec3(0.5));

	// calc reprojection
	vec4 reprojected_ndc = fogparam.last_viewproj*vec4(worldpos,1.0);
	vec4 reprojected_value = vec4(0.0);
	float reprojected_mix = 0.0;
	
#if 1
	if(reprojected_ndc.w>0) {
		reprojected_ndc/=reprojected_ndc.w;
		if (all(greaterThanEqual(reprojected_ndc.xyz, vec3(-1.0))) && all(lessThanEqual(reprojected_ndc.xyz, vec3(1.0)))  && reprojected_ndc.z >= 0.0) {
			// good reprojection
			reprojected_value = sample_volfog_volume(reprojected_ndc.xyz, previous_volume);
			worldpos = get_worldpos_new(fragcoords, invpixelsize, halton_map[int(fogparam.reprojection.x)]);
			reprojected_mix = 0.98;
		}
	}
#endif
	
	float falloff = fogparam.density_ani.z;
	float offset_ = fogparam.density_ani.w;
	float real_density  = fogparam.density_ani.x*min(exp(-(worldpos.y-offset_)*falloff),1.0);
	
	
	vec3 scattering_ = vec3(1.0)*real_density;
	
	vec3 inscattering = get_inscattering(worldpos);
	vec4 coloranddensity = vec4(inscattering*scattering_,real_density);
	//ComputeScattering(dot(raydir,-sun_dir));
	
	
	coloranddensity = mix(coloranddensity, reprojected_value, reprojected_mix);
	//coloranddensity.rgb = randColor(int(gl_GlobalInvocationID.z));
	//coloranddensity.a = 0.5;
	imageStore(VolumeTexture,ivec3(fragcoords),coloranddensity);//vec4(inscattering,density));
}
