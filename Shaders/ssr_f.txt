
layout(location = 0) in vec2 texCoord;
layout(location = 0) out vec4 FS_OUT_Color;

#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"
layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};
layout (binding = 7, std140) uniform SsrParamsUbo { SsrParams params; };
#include "commonUtil.txt"
#include "LightingShared.txt"

layout(binding=0) uniform sampler2D GBUFFER0;
layout(binding=1) uniform sampler2D GBUFFER1;
layout(binding=2) uniform sampler2D GBUFFER2;
layout(binding=3) uniform sampler2D DEPTH;
layout(binding=4) uniform sampler2D SceneColorMips;

vec2 RandN2(vec2 n)
{
	return fract(sin(dot(n, vec2(12.9898, 78.233))) * vec2(43758.5453123, 28001.8384));
}

vec3 TangentToWorld(vec3 N, vec3 H)
{
	vec3 upVector = abs(N.z) < 0.999 ? vec3(0.0, 0.0, 1.0) : vec3(1.0, 0.0, 0.0);
	vec3 T = normalize(cross(upVector, N));
	vec3 B = cross(N, T);
	return T * H.x + B * H.y + N * H.z;
}

vec4 ImportanceSampleGGX(vec2 e, float roughness)
{
	float m = roughness * roughness;
	float m2 = m * m;
	float phi = 2.0 * 3.14159265 * e.x;
	float cosTheta = sqrt((1.0 - e.y) / (1.0 + (m2 - 1.0) * e.y));
	float sinTheta = sqrt(1.0 - cosTheta * cosTheta);
	vec3 h;
	h.x = sinTheta * cos(phi);
	h.y = sinTheta * sin(phi);
	h.z = cosTheta;
	float r = (cosTheta * m2 - cosTheta) * cosTheta + 1.0;
	float d = m2 / (3.14159265 * r * r);
	return vec4(h, d * cosTheta);
}

// 0:1 UV from -1:1 clip
vec2 ClipToUv(vec2 clipPos)
{
	return clipPos * vec2(0.5, 0.5) + vec2(0.5, 0.5);
}

vec3 ProjectWorldToClip(vec3 wsPos)
{
	vec4 clipPos = g.viewproj * vec4(wsPos, 1.0);
	return clipPos.xyz / clipPos.w;
}

vec3 ProjectWorldToUv(vec3 wsPos)
{
	vec3 clipPos = ProjectWorldToClip(wsPos);
	return vec3(ClipToUv(clipPos.xy), clipPos.z);
}

float RayAttenBorder(vec2 pos, float value)
{
	float borderDist = min(1.0 - max(pos.x, pos.y), min(pos.x, pos.y));
	return clamp(borderDist > value ? 1.0 : borderDist / value, 0.0, 1.0);
}

void main()
{
	vec2 uv = texCoord;

	float depth = textureLod(DEPTH, uv, 0.0).r;
	vec4 metal_rough = textureLod(GBUFFER2, uv, 0.0);

	vec4 base = vec4(textureLod(SceneColorMips, uv, 0.0).rgb, 0.0);

	uint matType = get_gbuffer_material_type(metal_rough);
	float roughness = metal_rough.g;

	if (matType == MATERIAL_TYPE_UNLIT || roughness > params.roughnessFade || depth <= 0.0) {
		FS_OUT_Color = base;
		return;
	}

	vec3 worldPos = WorldPosFromDepth(uv, depth);
	vec3 worldNormal = textureLod(GBUFFER0, uv, 0.0).rgb;
	vec3 normalVS = normalize((g.view * vec4(worldNormal, 0.0)).xyz);
	vec3 viewPos = (g.view * vec4(worldPos, 1.0)).xyz;

	// OpenGL view space: -Z is forward. Reject reflections going toward camera (+Z)
	vec3 reflectVS = normalize(reflect(normalize(viewPos), normalVS));
	if (reflectVS.z > -0.001) {
		FS_OUT_Color = base;
		return;
	}

	// Stochastic reflection direction via GGX importance sampling
	vec3 camPos = g.viewpos_time.xyz;
	vec3 viewWS = normalize(worldPos - camPos);
	vec2 jitter = RandN2(uv + params.temporalTime);
	vec2 Xi = jitter;
	Xi.y = mix(Xi.y, 0.0, params.brdfBias);
	vec3 H = (params.temporalEffect > 0.5)
		? TangentToWorld(worldNormal, ImportanceSampleGGX(Xi, roughness).xyz)
		: worldNormal;
	vec3 reflectWS = reflect(viewWS, H);

	// Build ray path in UV space (z = raw depth)
	vec3 startWS = worldPos + worldNormal * params.worldAntiSelfOccBias;
	vec3 startUV = ProjectWorldToUv(startWS);

	// Compute reflection endpoint in view space, extend far for stable direction.
	// If endpoint goes behind camera (z > 0 in OpenGL), project onto z=-1 plane.
	vec3 reflDirVS = mat3(g.view) * reflectWS;
	vec3 reflEndVS = viewPos + reflDirVS * 1000.0;
	if (reflEndVS.z > 0.0)
		reflEndVS /= -reflEndVS.z;
	vec3 reflEndWS = (g.invview * vec4(reflEndVS, 1.0)).xyz;
	vec3 endUV = ProjectWorldToUv(reflEndWS);
	vec3 rayUV = endUV - startUV;
	vec2 rayUVAbs = abs(rayUV.xy);
	rayUV *= params.rayTraceStep / max(rayUVAbs.x, rayUVAbs.y);
	vec3 currOffset = startUV + rayUV * 2.0;
	vec3 rayStep = rayUV * 2.0;

	// Clamp sample count to screen edge
	vec3 samplesToEdge = ((sign(rayStep.xyz) * 0.5 + 0.5) - currOffset.xyz) / rayStep.xyz;
	samplesToEdge.x = min(samplesToEdge.x, min(samplesToEdge.y, samplesToEdge.z)) * 1.05;
	float numSamples = min(params.maxTraceSamples, samplesToEdge.x);
	rayStep *= samplesToEdge.x / numSamples;

	// Linear ray trace with binary refinement
	float depthDiffError = 1.3 * abs(rayStep.z);
	float currSampleIndex = 0.0;
	bool hit = false;
	while (currSampleIndex < numSamples)
	{
		float currSample = textureLod(DEPTH, currOffset.xy, 0.0).r;
		// Reverse-Z: closer = larger depth. Ray behind surface = ray depth < surface depth.
		float depthDiff = currSample - currOffset.z;

		if (depthDiff >= 0.0)
		{
			if (depthDiff < depthDiffError) {
				hit = true;
				break;
			}
			// Binary refinement: step back and halve
			currOffset -= rayStep;
			rayStep *= 0.5;
		}

		currOffset += rayStep;
		currSampleIndex++;
	}

	if (!hit) {
		FS_OUT_Color = base;
		return;
	}

	// Fade at screen edges
	const float fadeStart = 0.9;
	const float fadeEnd = 1.0;
	const float fadeDiffRcp = 1.0 / (fadeEnd - fadeStart);
	vec2 boundary = abs(currOffset.xy - vec2(0.5)) * 2.0;
	float fadeOnBorder = 1.0 - clamp((boundary.x - fadeStart) * fadeDiffRcp, 0.0, 1.0);
	fadeOnBorder *= 1.0 - clamp((boundary.y - fadeStart) * fadeDiffRcp, 0.0, 1.0);
	fadeOnBorder = smoothstep(0.0, 1.0, fadeOnBorder);
	fadeOnBorder *= RayAttenBorder(currOffset.xy, params.edgeFadeFactor);

	// Roughness fade
	float roughnessFade = clamp((params.roughnessFade - roughness) * 20.0, 0.0, 1.0);

	// Distance fade
	float viewDist = length(viewPos);
	float distanceFade = clamp((params.fadeOutDistance - viewDist) / params.fadeOutDistance, 0.0, 1.0);

	// Reflection angle fade (negate for OpenGL: -Z is forward)
	float reflectFade = clamp(-reflectVS.z * 8.0, 0.0, 1.0);

	float hitMask = fadeOnBorder * roughnessFade * distanceFade * reflectFade;

	// Sample color at hit with roughness-based mip (cone tracing)
	vec3 viewVector = -viewWS;
	float NdotV = clamp(dot(worldNormal, viewVector), 0.0, 1.0);
	float coneTangent = mix(0.0, roughness * 5.0 * (1.0 - params.brdfBias),
		pow(NdotV, 1.5) * sqrt(roughness));
	float intersectionCircleRadius = coneTangent * length(currOffset.xy - uv);
	float mip = clamp(log2(intersectionCircleRadius * params.traceSizeMax), 0.0, params.maxColorMiplevel);
	vec3 sampleColor = textureLod(SceneColorMips, currOffset.xy, mip).rgb;

	FS_OUT_Color = vec4(sampleColor, hitMask);
}
