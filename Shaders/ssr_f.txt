// SSR code from: https://sugulee.wordpress.com/2021/01/19/screen-space-reflections-implementation-and-optimization-part-2-hi-z-tracing-method/

float getMinimumDepthPlane(vec2 p, int mipLevel, sampler2D tex_hi_z)
{
	return textureLod(tex_hi_z,p,mipLevel).r;
}
vec3 intersectDepthPlane(vec3 o, vec3 d, float z)
{
	return o + d * z;
}
bool crossedCellBoundary(vec2 a, vec2 b)
{
	return any(notEqual(a, b));
}
vec2 getCellCount(int mipLevel, sampler2D tex_hi_z)
{
    return vec2(textureSize(tex_hi_z,mipLevel));
}
vec2 getCell(vec2 pos, vec2 cell_count)
{
    return vec2(floor(pos*cell_count));
}
vec3 intersectCellBoundary(vec3 o, vec3 d, vec2 cell, vec2 cell_count, vec2 crossStep, vec2 crossOffset)
{
	vec2 index_ = cell + crossStep;
	vec2 boundary = index_ / cell_count;
	boundary += crossOffset;

	vec2 delta = boundary - o.xy;
	delta /= d.xy;
	float t = min(delta.x, delta.y);

	return intersectDepthPlane(o, d, t);
}

float FindIntersection_HiZ(
	vec3 samplePosInTS,
	vec3 vReflDirInTS,
	float maxTraceDistance,
	sampler2D tex_hi_z,
	out vec3 intersection)
{
    const int maxLevel = textureQueryLevels(tex_hi_z);

    vec2 crossStep = vec2(vReflDirInTS.x>=0 ? 1 : -1, vReflDirInTS.y>=0 ? 1 : -1);
    vec2 crossOffset = crossStep / g.viewport_size.xy / 128;
    crossStep = clamp(crossStep,0.0,1.0);

    vec3 ray = samplePosInTS.xyz;
    float minZ = ray.z;
    float maxZ = ray.z + vReflDirInTS.z * maxTraceDistance;
    float deltaZ = (maxZ - minZ);	// can be negative

    vec3 o = ray;
    vec3 d = vReflDirInTS * maxTraceDistance;

    int startLevel = 2;
    int stopLevel = 0;
    vec2 startCellCount = getCellCount(startLevel, tex_hi_z);

    vec2 rayCell = getCell(ray.xy, startCellCount);
    ray = intersectCellBoundary(o, d, rayCell, startCellCount, crossStep, crossOffset*64);

    int level = startLevel;
    uint iter = 0;
    bool isBackwardRay = vReflDirInTS.z>0;
    float rayDir = isBackwardRay ? 1 : -1;

    while(level >= stopLevel && ray.z*rayDir <= maxZ*rayDir && iter < max_ssr_iters)
    {
		if(ray.x < 0.0 || ray.x > 1.0 || ray.y < 0.0 || ray.y > 1.0)
			return 0.0;
	
        const vec2 cellCount = getCellCount(level, tex_hi_z);
        const vec2 oldCellIdx = getCell(ray.xy, cellCount);

        float cell_minZ = getMinimumDepthPlane((oldCellIdx+0.5)/cellCount, level, tex_hi_z);
		
        vec3 tmpRay = ((cell_minZ < ray.z) && !isBackwardRay) ? 
			intersectDepthPlane(o, d, (cell_minZ - minZ)/deltaZ) : 
			ray;

        const vec2 newCellIdx = getCell(tmpRay.xy, cellCount);

        float thickness = level == 0 ? 
			(cell_minZ-ray.z) : 
			0;
        bool crossed = 
			(isBackwardRay && (cell_minZ < ray.z)) || 
			(thickness > MAX_THICKNESS) || 
			crossedCellBoundary(oldCellIdx, newCellIdx);
		
        ray = crossed ? 
			intersectCellBoundary(o, d, oldCellIdx, cellCount, crossStep, crossOffset) : 
			tmpRay;
        level = crossed ? 
			int(min(float(maxLevel), level + 1.0)) : 
			level - 1;

        ++iter;
    }
	
    intersection = ray;

    bool intersected = (level < stopLevel);

    return intersected ? 1.0 : 0.0;
}

vec4 ComputeReflectedColor(float intensity, vec3 intersection, vec4 skyColor, sampler2D sceneColor)
{
	if(intensity <= 0.0)
		return skyColor;

	vec2 uv = clamp(intersection.xy, 0.0, 1.0);
	vec4 scene = pow(texture(sceneColor, uv),vec4(1/2.2));
	return mix(skyColor, scene, intensity);
}
uniform mat4 g_proj;	// i dont store this in GlobalData

// Compute the position, the reflection direction, maxTraceDistance of the sample in texture space.
void ComputePosAndReflection(
	ivec2 texel_coord,
	vec2 uv,
	vec3 vSampleNormalInVS,
	sampler2D tex_depth,
	out vec3 outSamplePosInTS,
	out vec3 outReflDirInTS,
	out float outMaxDistance)
{
    float sampleDepth = texelFetch(tex_depth,texel_coord,0).r;
    vec4 samplePosInCS = vec4(uv * 2.0 - 1.0, sampleDepth, 1.0);

    vec4 samplePosInVS = g.invproj * samplePosInCS;
    samplePosInVS /= samplePosInVS.w;

    vec3 vCamToSampleInVS = normalize(samplePosInVS.xyz);
    vec4 vReflectionInVS = vec4(reflect(vCamToSampleInVS.xyz, vSampleNormalInVS.xyz), 0.0);

    vec4 vReflectionEndPosInVS = samplePosInVS + vReflectionInVS * 1000.0;
	vReflectionEndPosInVS /= (vReflectionEndPosInVS.z > 0 ? (-vReflectionEndPosInVS.z) : 1);
    vec4 vReflectionEndPosInCS = g_proj * vec4(vReflectionEndPosInVS.xyz, 1.0);
    vReflectionEndPosInCS /= vReflectionEndPosInCS.w;
    vec3 vReflectionDir = normalize((vReflectionEndPosInCS - samplePosInCS).xyz);

    // Transform to texture space
    samplePosInCS.xy *= vec2(0.5, 0.5);
    samplePosInCS.xy += vec2(0.5, 0.5);

    vReflectionDir.xy *= vec2(0.5, 0.5);

    outSamplePosInTS = vec3(samplePosInCS.xy,sampleDepth);
    outReflDirInTS = vReflectionDir.xyz;

    // Compute the maximum distance to trace before the ray goes outside of the visible area.
    outMaxDistance = outReflDirInTS.x >= 0.0 ? (1.0 - outSamplePosInTS.x) / outReflDirInTS.x : -outSamplePosInTS.x / outReflDirInTS.x;
    outMaxDistance = min(outMaxDistance, outReflDirInTS.y < 0.0 ? (-outSamplePosInTS.y / outReflDirInTS.y) : ((1.0 - outSamplePosInTS.y) / outReflDirInTS.y));
	
    outMaxDistance = min(outMaxDistance, 
		outReflDirInTS.z < 0.0 ? 
			(-outSamplePosInTS.z / outReflDirInTS.z) : 
			((1.0 - outSamplePosInTS.z) / outReflDirInTS.z)
	);
}

float Halton(int index_, int base)
{
    float f = 1.0;
    float r = 0.0;

    int i = index_;
    while(i > 0)
    {
        f = f / float(base);
        r = r + f * float(i % base);
        i = i / base;
    }

    return r;
}
vec2 Halton2D(int index_)
{
    return vec2(
        Halton(index_, 2),
        Halton(index_, 3)
    );
}
// from flax
vec2 RandN2(vec2 n)
{
    return fract(sin(dot(n, vec2(12.9898, 78.233))) * vec2(43758.5453123, 28001.8384));
}
vec3 TangentToWorld(vec3 vec, vec3 tangentZ)
{
    vec3 upVector = abs(tangentZ.z) < 0.999 ? vec3(0, 0, 1) : vec3(1, 0, 0);
    vec3 tangentX = normalize(cross(upVector, tangentZ));
    vec3 tangentY = cross(tangentZ, tangentX);
    return tangentX * vec.x + tangentY * vec.y + tangentZ * vec.z;
}
vec4 ImportanceSampleGGX(vec2 e, float roughness)
{
    float m = roughness * roughness;
    float m2 = m * m;

    float phi = 2 * 3.14159 * e.x;
    float cosTheta = sqrt((1 - e.y) / (1 + (m2 - 1) * e.y));
    float sinTheta = sqrt(1 - cosTheta * cosTheta);

    vec3 h;
    h.x = sinTheta * cos(phi);
    h.y = sinTheta * sin(phi);
    h.z = cosTheta;

    float r = (cosTheta * m2 - cosTheta) * cosTheta + 1;
    float d = m2 / (3.14159 * r * r);
    float pdf = d * cosTheta;
    return vec4(h, pdf);
}

vec2 HaltonJittered(vec2 uv, int frame)
{
    vec2 h = Halton2D(frame);
    vec2 jitter = RandN2(uv);

    return fract(h + jitter);
}
vec4 ScreenSpaceReflectionDirection(vec2 uv, vec3 world_normal, float roughness, float brdfBias, int temp_add)
{
    // Randomize it a little
    //vec2 jitter = RandN2(uv + vec2(temporalTime));
	
	int sampleIndex = int(temporalTime + temp_add)%1024;
	vec2 jitter = HaltonJittered(uv,sampleIndex);
	
    vec2 Xi = jitter;
    Xi.y = mix(Xi.y, 0.0, brdfBias);
	vec4 t = ImportanceSampleGGX(Xi, roughness);
    vec3 H =TangentToWorld(t.xyz,world_normal);
	return vec4(H,t.w);
}

vec3 do_raymarch(vec2 uv, vec3 reflDir_1_VS, float roughness, vec3 viewPos, vec3 normalVS, int rayIndex)
{
	
	vec4 refl_and_pdf = ScreenSpaceReflectionDirection(uv,reflDir_1_VS,roughness,0.9,rayIndex);
	vec3 reflDirVS = refl_and_pdf.xyz;
	//vec3 normalVSRough = (g.view * vec4(worldH, 0.0)).xyz;
	
	//vec3 reflDirVS = (g.view * vec4(worldReflect, 0.0)).xyz;

	
    // View direction (camera at origin)
	//vec3 reflDirVS = reflect(-viewDir, viewH);

    // Reflection direction in view space

    // Initialize ray in view space
	//float step_sz_to_use = step_size * (max(dot(normalVS, reflDirVS), 0.2)*2.0);
    vec3 rayPosVS = viewPos + normalVS * step_size * bias;
	float step_sz_to_use = step_size;// * length(rayPosVS - viewPos) / max(dot(normalVS, reflDirVS), 0.1);
vec3 col = vec3(0.0);
    //vec3 prevPos = rayPosVS;

    for (int i = 0; i < MAX_STEPS; i++)
    {
        rayPosVS += reflDirVS * step_sz_to_use;

        if (length(rayPosVS) > max_distance)
            break;

        // Project ray to screen
        vec4 rayClip = g_proj * vec4(rayPosVS, 1.0);
        vec2 rayUV = rayClip.xy / rayClip.w * 0.5 + 0.5;

        // Edge fading
        float edgeFade = smoothstep(0.0, 0.05, min(rayUV.x, rayUV.y)) *
                         smoothstep(1.0, 0.95, max(rayUV.x, rayUV.y));

        if (rayUV.x < 0.0 || rayUV.x > 1.0 || rayUV.y < 0.0 || rayUV.y > 1.0)
            break;

        float depthSample = textureLod(DEPTH, rayUV, 0.0).r;
        if (depthSample <= 0.0)
            continue;

        // Linearize depth
        float linearSceneZ = -g.near / depthSample; // Replace with proper linearization if needed

        // Adaptive thickness
        float thickness = max_thickness ;/// max(dot(normalVS, reflDirVS), 0.4);

        // Check hit
		float depth_diff = (rayPosVS.z - linearSceneZ);
		
        if (depth_diff<0)	// my pos is behind geometry
        {
			rayPosVS -= reflDirVS * step_sz_to_use;	// set to previous pos
			step_sz_to_use *= 0.5;	// half step size on next loop
		
			if(abs(depth_diff)<=thickness) {
				//vec2 sample_pos = (rayUV*vec2(g.viewport_size.xy)+vec2(0.5))/g.viewport_size.xy;
				col=vec3(rayUV, refl_and_pdf.w);//
				//if(debug_toggle)
				//	col =texelFetch(SceneColor, ivec2(rayUV*g.viewport_size.xy), 0).rgb;// * edgeFade;
				break;
            }
        }

        //prevPos = rayPosVS;

   // col = mix(albedo, col, 0.5);//pow(mix(albedo, col, 0.5), vec3(1.0/2.2));
