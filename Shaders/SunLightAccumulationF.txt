vec3 pbr_calculate_sun_disk(vec4 sunDir4, vec4 sunCol4, vec3 worldspace, vec3 Albedo, float Roughness, float Metalness, vec3 Normal)
{
	vec3 LightDir = -sunDir4.xyz;
	vec3 LightColor = sunCol4.xyz;
	float sinAlpha = sunDir4.w;
	float cosAlpha = sunCol4.w;

	vec3 view_pos = g.viewpos_time.xyz;
	vec3 V = normalize(view_pos - worldspace);
	vec3 R = reflect(-V, Normal);
	vec3 F0 = mix(vec3(0.04), Albedo, Metalness);

	float NdotL = max(dot(Normal, LightDir), 0.0);

	// diffuse: punctual
	vec3 kS_diffuse = fresnelSchlick(max(dot(normalize(V + LightDir), V), 0.0), F0);
	vec3 kD = (vec3(1.0) - kS_diffuse) * (1.0 - Metalness);
	vec3 diffuse = kD * Albedo / PI * LightColor * NdotL;

	// specular: representative point on disk
	float DdotR = dot(LightDir, R);
	vec3 S = R - DdotR * LightDir;
	float S2 = dot(S, S);
	vec3 Lspec = LightDir;
	if (DdotR >= cosAlpha)
		Lspec = R;
	else if (S2 > 1e-8)
		Lspec = normalize(cosAlpha * LightDir + (S / sqrt(S2)) * sinAlpha);

	vec3 Hspec = normalize(V + Lspec);
	float NdotLspec = max(dot(Normal, Lspec), 0.0);
	float NDF = DistributionGGX(Normal, Hspec, Roughness);
	float G   = GeometrySmith(Normal, V, Lspec, Roughness);
	vec3 F    = fresnelSchlick(max(dot(Hspec, V), 0.0), F0);
	vec3 specular = (NDF * G * F) / (4.0 * max(dot(Normal, V), 0.0) * NdotLspec + 0.0001);

	return diffuse + specular * LightColor * NdotLspec;
}

const int CASCADES_USED = 3;	// change this in C++ also


float InterleavedGradientNoise(vec2 pixel)
{
    return fract(52.9829189 * fract(dot(pixel, vec2(0.06711056, 0.00583715))));
}

float linear_step(float edge0, float edge1, float x)
{
	float size = edge1-edge0;
	return clamp((x-edge0)/size,0.0,1.0);
}

float sample_sun_shadow(vec3 WorldSpace,float linear_depth
#ifdef DEBUG
,out int OUTlayer
#endif
)
{
#ifdef DEBUG
	OUTlayer = -1;
#endif

	int layer = -1;
	int layer2 = -1;	//overlapping
	float lerp_val=1.0;
	for(int i=0;i<CASCADES_USED;i++) {
	// This adds some instructions but it looks so much better than just checking the plane distance since cascades overlap
		vec4 frag_pos_light_space = CascadeMats.Mat[i]*vec4(WorldSpace,1.0);
		vec3 proj_coords=frag_pos_light_space.xyz/frag_pos_light_space.w;
		if(proj_coords.x>-0.99&&proj_coords.x<0.99&&proj_coords.y>-0.99&&proj_coords.y<0.99 
		&& proj_coords.z > 0.001 && proj_coords.z < 0.99 // 7/30: reverse z change
		
		) {
			if(layer==-1) {
				layer=i;
				vec3 absp = abs(proj_coords);
				float mv = max(absp.z, max(absp.x,absp.y));
				lerp_val = linear_step(0.9,1.0,mv);
			}
			else {
				layer2 = i;
				break;
			
			}
		
		}
	}
	if(layer==-1)
		return 1.0;

	
	float splitStart = CascadeMats.Splits[layer];
	float splitEnd   = CascadeMats.Splits[layer + 1];
	float blend_range = (splitEnd-splitStart)*0.1;
	float blend = clamp(
		(linear_depth - (splitEnd - blend_range)) / blend_range,
		0.0, 1.0
	);
	

	float noise = InterleavedGradientNoise(gl_FragCoord.xy);
	if (noise < lerp_val)
		layer = layer2;
	if(layer==-1)
		return 1.0;
		
	

	vec4 frag_pos_light_space = CascadeMats.Mat[layer]*vec4(WorldSpace,1.0);
	vec3 proj_coords=frag_pos_light_space.xyz/frag_pos_light_space.w;
	proj_coords.xy=proj_coords.xy*0.5+0.5;	//7/30: reverse z change
	float depth_light_space=proj_coords.z;
	if(depth_light_space>1.05)
		return 1.0;
	
#ifdef DEBUG
	OUTlayer = layer;
#endif	
	float bias = uEpsilon;
	
	float shadow = 0.0;
	vec2 texelSize = 1.0 / vec2(textureSize(cascade_shadow_map, 0));
	vec2 grad = fract(proj_coords.xy/texelSize);
	for(int x = -1; x <= 1; ++x)
	{
		for(int y = -1; y <=1; ++y)
		{
			#ifdef CHEAP_PCF
		
				float pcf_depth = texture(
							cascade_shadow_map,
							vec3(proj_coords.xy + vec2(x, y) * texelSize,
							layer)
							).r; 
				shadow += (depth_light_space - bias) > pcf_depth ? 1.0 : 0.0;
			#else	// Bilinear filtered, looks nice
					shadow += texture(cascade_shadow_map,vec4(proj_coords.xy + vec2(x, y) * texelSize,
							layer,(depth_light_space-bias)));
			#endif
		}    
	}
	shadow /= 9.0;

	return 1.0-shadow;
}

#ifdef CONTACT_SHADOWS
#define contactShadowMaxDist   params.contactShadowParams.x
#define contactShadowThickness params.contactShadowParams.y
#define contactShadowSteps     params.contactShadowParams.z
#define contactShadowIntensity params.contactShadowParams.w

// Short fixed-length screen-space ray march toward the sun, catching near-camera
// contact occlusion (thin/small geometry) that cascade shadow map resolution misses.
// Mirrors the ssr_f.txt raymarch pattern but marches a fixed short world-space
// distance toward the light instead of a long variable-length reflection ray, so no
// binary refinement is needed.
float sample_contact_shadow(vec3 worldspace, vec3 normalWS)
{
	vec3 lightDirWS = -params.uSunDirection.xyz; // points from surface toward the sun

	vec3 startWS = worldspace + normalWS * contactShadowThickness;
	vec3 endWS = startWS + lightDirWS * contactShadowMaxDist;

	vec3 startUV = ProjectWorldToUv(startWS);
	vec3 endUV = ProjectWorldToUv(endWS);

	int steps = int(contactShadowSteps);
	vec3 rayStep = (endUV - startUV) / float(steps);

	// Reverse-Z: depth values shrink with distance from the ray-start plane, so scale
	// the thickness window by the per-step depth delta to reject far-behind false hits.
	float depthThickness = abs(rayStep.z) * 2.0 + 0.0001;

	vec3 curr = startUV + rayStep;
	float occlusion = 0.0;
	for (int i = 0; i < steps; i++)
	{
		if (curr.x < 0.0 || curr.x > 1.0 || curr.y < 0.0 || curr.y > 1.0)
			break;

		float sampledDepth = textureLod(DEPTH, curr.xy, 0.0).r;
		float depthDiff = sampledDepth - curr.z;
		if (depthDiff >= 0.0 && depthDiff < depthThickness)
		{
			occlusion = 1.0;
			break;
		}

		curr += rayStep;
	}

	return 1.0 - occlusion * contactShadowIntensity;
}
#endif

void main()
{
	vec2 uv = vec2(gl_FragCoord.xy) / vec2(g.viewport_size.xy);
	vec4 metal_rough = texture(GBUFFER2,uv);
	vec4 albedo = texture(GBUFFER1,uv);
	vec4 depth_tex = texture(DEPTH,uv);
	vec4 normal_t = texture(GBUFFER0,uv);
	
	uint type = get_gbuffer_material_type(metal_rough);
	if(type==MATERIAL_TYPE_UNLIT)
		discard;
	
	float depth_value = depth_tex.r;
	vec3 worldspace = WorldPosFromDepth(uv, depth_value);
	float linear_depth = g.near/depth_value;
	#ifdef DEBUG
		int layer = 0;
		float shadow_factor = sample_sun_shadow(worldspace,linear_depth,layer);
		if(layer==0)
			FS_OUT_Color = vec4(1,0,0,1);
		else if(layer==1)
			FS_OUT_Color = vec4(0,1,0,1);
		else if(layer==2)
			FS_OUT_Color = vec4(0,0,1,1);
		else if(layer==3)
			FS_OUT_Color = vec4(1,1,0,1);
		else if(layer == -1)
			FS_OUT_Color = vec4(0,0,0,1);
		
			
		//FS_OUT_Color = vec4(shadow_factor);
	#else
		float shadow_factor = sample_sun_shadow(worldspace,linear_depth);
		#ifdef CONTACT_SHADOWS
			shadow_factor = min(shadow_factor, sample_contact_shadow(worldspace, normal_t.xyz));
		#endif
		vec3 out_color = pbr_calculate_sun_disk(params.uSunDirection, params.uSunColor, worldspace, albedo.xyz, metal_rough.y, metal_rough.x, normal_t.xyz);
		out_color = apply_material_directional(type, out_color, normal_t.xyz, worldspace,
		                                        -params.uSunDirection.xyz, params.uSunColor.xyz,
		                                        albedo.xyz, metal_rough.y, metal_rough.x,
		                                        get_gbuffer_alias0(metal_rough), get_gbuffer_alias1(albedo));
		FS_OUT_Color = vec4(shadow_factor*out_color,1.0);
	#endif
}
