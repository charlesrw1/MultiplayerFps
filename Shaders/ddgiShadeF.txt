vec3 WorldPosFromDepth(vec2 TexCoord, float depth) {

	// 7/30: reverse Z change
    float z = depth;// * 2.0 - 1.0;

    vec4 clipSpacePosition = vec4(TexCoord * 2.0 - 1.0, z, 1.0);
    vec4 viewSpacePosition = g.inv_viewproj * clipSpacePosition;

    // Perspective division
    viewSpacePosition /= viewSpacePosition.w;

    return viewSpacePosition.xyz;
}

vec3 get_probe_world_from_coords(ivec3 coords)
{
return  volume_origin+vec3(coords)*volume_spacing;
}

vec2 OctahedralEncode(vec3 normal_in)
{
	vec3 normal = normal_in.xzy;

	normal = normalize(normal);
	float l1norm = abs(normal.x) + abs(normal.y) + abs(normal.z);
	vec2 result = normal.xy * (1.0 / l1norm);
	
	if (normal.z < 0.0) {
		vec2 oldResult = result;
		result.x = (1.0 - abs(oldResult.y)) * (oldResult.x >= 0.0 ? 1.0 : -1.0);
		result.y = (1.0 - abs(oldResult.x)) * (oldResult.y >= 0.0 ? 1.0 : -1.0);
	}
	
	return result * 0.5 + 0.5; // Map to [0,1]
}

vec2 get_depth_probe_uv(vec2 octant_uv, ivec3 probe_coord) {
	const int DEPTH_IN_SIZE = 14;
	const int DEPTH_TILE_SIZE = 16;
	
	vec2 coords;
	coords.x = probe_coord.x*DEPTH_TILE_SIZE + (vol_grid.x*probe_coord.y*DEPTH_TILE_SIZE);
	coords.y = probe_coord.z*DEPTH_TILE_SIZE;
	
	coords += vec2(octant_uv*vec2(DEPTH_IN_SIZE))+vec2(1);
	vec2 final_uv = vec2(coords)/vec2(vol_grid.x*vol_grid.y*DEPTH_TILE_SIZE,vol_grid.z*DEPTH_TILE_SIZE);

	return final_uv;
}

vec2 get_irrad_probe_uv(vec2 octant_uv, ivec3 probe_coord) {
	const int DEPTH_IN_SIZE = 6;
	const int DEPTH_TILE_SIZE = 8;
	
	vec2 coords;
	coords.x = probe_coord.x*DEPTH_TILE_SIZE + (vol_grid.x*probe_coord.y*DEPTH_TILE_SIZE);
	coords.y = probe_coord.z*DEPTH_TILE_SIZE;
	
	coords += vec2(octant_uv*vec2(DEPTH_IN_SIZE))+vec2(1);
	vec2 final_uv = vec2(coords)/vec2(vol_grid.x*vol_grid.y*DEPTH_TILE_SIZE,vol_grid.z*DEPTH_TILE_SIZE);

	return final_uv;
}

struct OutIrrad {
	vec3 irrad;
	int closest_probe;
};
OutIrrad calculate_irradiance(vec3 direction, vec3 worldspace)
{
	
		
	vec3 biased_world_position = worldspace+direction*0.1-normalize(worldspace-g.viewpos_time.xyz)*0.1;
	
	ivec3 base_probe_coords = ivec3((biased_world_position-volume_origin)/volume_spacing);
	base_probe_coords = clamp(base_probe_coords,ivec3(0), 
	ivec3(vol_grid)-ivec3(1)
	);
	vec3 base_probe_pos = get_probe_world_from_coords(base_probe_coords);
	
	vec3 alpha = clamp(
		(biased_world_position-base_probe_pos)/volume_spacing,
		vec3(0),
		vec3(1));
	

	vec3 irradiance=vec3(0.0);
	float accum_weight = 0.0;
	float cheb_weight_total = 0.0;
	
	int best_probe = 0;
	float best_weight = 0.0;

	for(int probeIndex = 0;probeIndex < 8; probeIndex++) {
		ivec3 adjacent_probe_ofs = ivec3(probeIndex, probeIndex>>1, probeIndex>>2)&ivec3(1);
		ivec3 adjacent_probe_coords = clamp(
			base_probe_coords+adjacent_probe_ofs,
			ivec3(0),
			ivec3(vol_grid)-ivec3(1));
		vec3 adj_probe_worldpos = get_probe_world_from_coords(adjacent_probe_coords);
		

		vec3 biased_pos_to_adj_probe = normalize(adj_probe_worldpos-biased_world_position);
		float biased_pos_to_adj_probe_len = length(adj_probe_worldpos-biased_world_position);
		
		vec3 trilinear = max(vec3(0.001), 
			mix(vec3(1.0-alpha),vec3(alpha),vec3(adjacent_probe_ofs)));
		float trilinear_weight = trilinear.x*trilinear.y*trilinear.z;
		float weight = 1.0;
		float warp_shading = dot(biased_pos_to_adj_probe,direction)*0.5 + 0.5;
		weight *= (warp_shading*warp_shading)+0.2;
		
		vec2 octant_coords = OctahedralEncode(-biased_pos_to_adj_probe);
		
		vec2 probe_texture_uv = get_depth_probe_uv(
		octant_coords,
		adjacent_probe_coords);
		
		vec2 filtered_distance = vec2(2.0)*texture(probe_depth, probe_texture_uv).rg;
		
		if(filtered_distance.x<0.000001)
			continue;
		
		float variance = abs(
		(filtered_distance.x*filtered_distance.x) - filtered_distance.y);
		
		float chebyshev_weight = 1.0;
		if(biased_pos_to_adj_probe_len > filtered_distance.x) {
			float v = biased_pos_to_adj_probe_len - filtered_distance.x;
			chebyshev_weight = variance/(variance+(v*v));
			chebyshev_weight = max(
			chebyshev_weight*chebyshev_weight*chebyshev_weight,
			0.0
			);
		}
		
		weight *= max(0.05, chebyshev_weight);
		weight = max(0.000001, weight);
		const float crush_thresh = 0.2;
		if(weight<crush_thresh) {
			weight *= (weight*weight)*(1.0/(crush_thresh*crush_thresh));
		}
		weight *= trilinear_weight;
		
		octant_coords = OctahedralEncode(direction);
		probe_texture_uv = get_irrad_probe_uv(octant_coords,adjacent_probe_coords);
		
		vec3 probe_irrad = texture(probe_irradiance, probe_texture_uv).rgb;
		//probe_irrad = pow(probe_irrad, vec3(2.5));
		
		// approximate specular?
		
		
		
		irradiance += (weight*probe_irrad);
		accum_weight += weight;
		// holy crap so much code!
		
		if(weight>best_weight) {
			best_probe = adjacent_probe_coords.x + adjacent_probe_coords.y*vol_grid.x + adjacent_probe_coords.z*vol_grid.x*vol_grid.y;
			best_weight = weight;
			}
		
		//cheb_weight_total+= chebyshev_weight;
	}
	
	irradiance *= (1.0/accum_weight);
	
	OutIrrad outd;
	outd.irrad = irradiance;
	outd.closest_probe =best_probe;// BestCubemap.data[best_probe];
	
	//randColor(BestCubemap.data[best_probe]+1);
	return outd;
}

vec3 FresnelSchlickRougness(float cosTheta, vec3 F0, float roughness)
{
	 return F0 + (max(vec3(1.0 - roughness), F0) - F0) * pow(clamp(1.0 - cosTheta, 0.0, 1.0), 5.0);
}

vec3 get_albedo_mult(vec3 worldspace, vec3 Albedo, float Metalness, float Roughness, vec3 Normal)
{
	vec3 view_pos = g.viewpos_time.xyz;
	vec3 view_front = g.viewfront.xyz;

	vec3 V = normalize(view_pos-worldspace);
	
	vec3 F0 = vec3(0.04);
	F0 = mix(F0,Albedo,Metalness);
	
	vec3 F = FresnelSchlickRougness(max(dot(Normal,V),0.0),F0,Roughness);
	
	vec3 kS = F;
	vec3 kD = (1.0-kS)*(1.0-Metalness);
	
	return (kD*Albedo);
}

float compute_specular_occlusion(float ao, float NdotV, float roughness)
{
    float out_val = (pow(NdotV + ao, exp2(-16.0 * roughness - 1.0)) - 1.0 + ao);
	return clamp(out_val,0,1);
}

vec3 get_prefiltered_color(vec3 worldpos, vec3 R, float Roughness)
{
	float weight_sum = 0.0;
	vec3 sum = vec3(0.0);
			const float MAX_REFLECTION_LOD = 7.0;

	for(int i=0;i<num_cubemaps;i++) {
		vec4 pos_radius = Cubemaps.data[i];
		float dist = length(worldpos-pos_radius.xyz);
		float alpha = 1.0-smoothstep(pos_radius.w, pos_radius.w+2.0, dist);
		vec3 pf = textureLod(probes,vec4(R,float(i)),Roughness*MAX_REFLECTION_LOD).rgb;
		sum +=pf*alpha;
		weight_sum += alpha;
	}
	
	if(weight_sum <= 0.00001) {
		return textureLod(probes,vec4(R,float(0)),Roughness*MAX_REFLECTION_LOD).rgb;
	}
	return sum / weight_sum;
}

vec3 pbr_calculate_reflection(vec3 worldspace, vec3 Albedo, float Roughness, float Metalness, vec3 Normal, float ambient_occlusion, int probe_num)
{
	vec3 view_pos = g.viewpos_time.xyz;
	vec3 view_front = g.viewfront.xyz;

	vec3 V = normalize(view_pos-worldspace);
	vec3 R = reflect(-V,Normal);
	
	vec3 F0 = vec3(0.04);
	F0 = mix(F0,Albedo,Metalness);
	
	const float MAX_REFLECTION_LOD = 7.0;
	//vec3 prefiltered_color = textureLod(Skylight,vec3(R), Roughness*MAX_REFLECTION_LOD).rgb;

	vec3 prefiltered_color =get_prefiltered_color(worldspace,R,Roughness);// textureLod(probes,vec4(R,float(probe_num)),Roughness*MAX_REFLECTION_LOD).rgb;

	float NdotV = max(dot(Normal,V),0.0);
	vec3 F = FresnelSchlickRougness(NdotV,F0,Roughness);
	//float F90 = clamp(50*dot(F0,vec3(0.33)),0,1);
	//vec3 F = modified_f_schlick(F0,F90,NdotV);
	
	vec2 env_brdf = texture(PBR_brdflut,vec2(max(dot(Normal,V),0.0),Roughness)).rg;
	
	vec3 specular = prefiltered_color*(F*env_brdf.x+env_brdf.y);
	float spec_occlusion =compute_specular_occlusion(ambient_occlusion,NdotV,Roughness);
	

	return specular*spec_occlusion;
}
