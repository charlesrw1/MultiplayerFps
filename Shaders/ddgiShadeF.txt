#define halfres_texel_offset params.halfres_texel_offset


void main()
{
	ivec2 texel_to_fetch = ivec2(gl_FragCoord.xy);
	
#ifdef HALFRES_DDGI
	{
		texel_to_fetch *= 2;	// 2x2
		texel_to_fetch += halfres_texel_offset;	// for temporal accum
	}
#endif
	vec2 uv = vec2(vec2(texel_to_fetch)+vec2(0.5)) / vec2(g.viewport_size.xy);
	
	vec4 depth_tex = texelFetch(DEPTH,texel_to_fetch,0);
	vec4 normal_t = texelFetch(GBUFFER0,texel_to_fetch,0);
	vec4 metal_rough = texelFetch(GBUFFER2,texel_to_fetch,0);
	
	
	uint type = get_gbuffer_material_type(metal_rough);
	if(type==MATERIAL_TYPE_UNLIT)
		discard;
		
	vec3 direction = normal_t.xyz;
	
	float depth_value = depth_tex.r;
	vec3 worldspace = WorldPosFromDepth(uv, depth_value);
	


	
	//vec2 uv_gi = (worldspace.xz+vec2(10.0))/vec2(20.0);
//	uv_gi = clamp(uv_gi,vec2(0),vec2(1));
	
	//vec3 weird_gi_top = texture(top_image,uv_gi).rgb;
	//vec3 weird_gi_bot = texture(bottom_image,uv_gi).rgb;
	
	//float alpha = direction.y*0.5 + 0.5;
	//vec3 weird_gi = mix(weird_gi_top,weird_gi_bot,1-alpha);
	// intensifying the ao here. i guess this isnt "right". but otherwise specular occlusion is too weak in my opinion
	

	vec3 biased_world_position = worldspace+direction*ddgi_global.normal_bias-normalize(worldspace-g.viewpos_time.xyz)*ddgi_global.view_bias;
	vec3 outirrad;
	if(!calculate_irradiance(direction, worldspace, biased_world_position, outirrad)) {
		// no DDGI volume covers this point (none baked, or outside all volumes) - use sky ambient.
		// scaled by PI to match calculate_irradiance's normalization, since downstream code
		// multiplies outirrad by albedo/ssao without re-applying that factor.
		outirrad = mix(ground_color, sky_color, direction.y*0.5+0.5) * PI;
	}
	//outirrad.irrad = vec3(0.0);
	
#ifdef HALFRES_DDGI
	FS_OUT_Color = vec4(outirrad,1);
#else
	vec4 albedo = texelFetch(GBUFFER1,texel_to_fetch,0);
	float ssao_value = texture(SSAOBlurred, uv).r;
	vec3 albedo_mult = pbr_get_albedo_mult(worldspace, albedo.rgb, metal_rough.y, metal_rough.x, direction);
	vec3 rad = outirrad * albedo_mult * ssao_value;
	float alias0 = get_gbuffer_alias0(metal_rough);
	float alias1 = get_gbuffer_alias1(albedo);
	rad *= apply_material_diffuse_atten(type, direction, worldspace, alias0, alias1);

	rad = apply_specular_reflections(rad, worldspace, direction, albedo, metal_rough, type, alias0, alias1,
		ssao_value, uv, outirrad, include_cubemaps, specular_ao_intensity);

	FS_OUT_Color = vec4(rad, 1);
#endif

}
