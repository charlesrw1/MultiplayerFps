ivec3 get_actual_probe_coord(ivec3 input_)
{
	int index_ = input_.x + vol_grid.x*input_.y + vol_grid.x*vol_grid.y*input_.z;
	
	int probe_number = ProbeIndirection.data[index_];

	int probe_x = probe_number % int(vol_grid.x);
	int probe_y = (probe_number / int(vol_grid.x)) % int(vol_grid.y);
	int probe_z = probe_number / (int(vol_grid.x) * int(vol_grid.y));
	
	return ivec3(probe_x,probe_y,probe_z);
}



vec3 debug_one_probe(vec3 dir, ivec3 probeCoord)
{
	const int TILE_SIZE = 8;
	vec2 uv = OctahedralEncode(normalize(dir));
	
	vec2 coords;
	coords.x = probeCoord.x*TILE_SIZE + (vol_grid.x*probeCoord.y*TILE_SIZE);
	coords.y = probeCoord.z*TILE_SIZE;
	
	coords += vec2(uv*vec2(6))+vec2(1);
	vec2 final_uv = vec2(coords)/vec2(vol_grid.x*vol_grid.y*TILE_SIZE,vol_grid.z*TILE_SIZE);
	
	vec3 outc = texture(probe_irradiance,final_uv).rgb;

	return outc;


}



float compute_specular_occlusion(float ao, float NdotV, float roughness)
{
    float out_val = (pow(NdotV + ao, exp2(-16.0 * roughness - 1.0)) - 1.0 + ao);
	return clamp(out_val,0,1);
}

vec3 get_prefiltered_color(vec3 worldpos, vec3 R, float Roughness)
{
	float weight_sum = 0.0;
	vec3 sum = vec3(0.0);
			const float MAX_REFLECTION_LOD = 7.0;

	for(int i=0;i<num_cubemaps;i++) {
		vec4 pos_radius = Cubemaps.data[i];
		float dist = length(worldpos-pos_radius.xyz);
		float alpha = 1.0-smoothstep(pos_radius.w, pos_radius.w+2.0, dist);
		vec3 pf = textureLod(probes,vec4(R,float(i)),Roughness*MAX_REFLECTION_LOD).rgb;
		sum +=pf*alpha;
		weight_sum += alpha;
	}
	
	if(weight_sum <= 0.00001) {
		return textureLod(probes,vec4(R,float(0)),Roughness*MAX_REFLECTION_LOD).rgb;
	}
	return sum / weight_sum;
}

vec3 pbr_calculate_reflection(vec3 worldspace, vec3 Albedo, float Roughness, float Metalness, vec3 Normal, float ambient_occlusion, vec3 ref_irrad)
{
	vec3 view_pos = g.viewpos_time.xyz;
	vec3 view_front = g.viewfront.xyz;

	vec3 V = normalize(view_pos-worldspace);
	vec3 R = reflect(-V,Normal);
	
	vec3 F0 = vec3(0.04);
	F0 = mix(F0,Albedo,Metalness);
	
	const float MAX_REFLECTION_LOD = 7.0;
	//vec3 prefiltered_color = textureLod(skyprobe,vec3(R), Roughness*MAX_REFLECTION_LOD).rgb;
	//vec3 prefiltered_color = pow(ref_irrad,vec3(1.5));
	vec3 prefiltered_color = get_prefiltered_color(worldspace,R,Roughness);///get_prefiltered_color(worldspace,R,1.0);// textureLod(probes,vec4(R,float(probe_num)),Roughness*MAX_REFLECTION_LOD).rgb;
	//prefiltered_color *= ref_irrad;

	float NdotV = max(dot(Normal,V),0.0);
	vec3 F = FresnelSchlickRougness(NdotV,F0,Roughness);
	//float F90 = clamp(50*dot(F0,vec3(0.33)),0,1);
	//vec3 F = modified_f_schlick(F0,F90,NdotV);
	
	vec2 env_brdf = texture(PBR_brdflut,vec2(max(dot(Normal,V),0.0),Roughness)).rg;
	
	vec3 specular = prefiltered_color*(F*env_brdf.x+env_brdf.y);
	float spec_occlusion =compute_specular_occlusion(ambient_occlusion,NdotV,Roughness);
	

	return specular*spec_occlusion;
}
