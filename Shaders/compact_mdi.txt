
layout (local_size_x = 32) in;

#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"
layout (binding = 7, std140) uniform CullParamsUbo { CullParams params; };

// draw calls
layout(binding = 2, std430) buffer DrawElementsIndirectCommand_Buffer {
	DrawElementsIndirectCommand draw_cmd_out[];
};

layout(binding = 3, std430) readonly buffer multidraw_batches_buffer {
	uint multidraw_batches[];
};

layout(binding = 5, std430)  buffer counts_buffer {
	uint counts[];
};
layout(binding = 6, std430) readonly buffer draw_to_batch_buffer {
	int draw_to_batch[];
};

void main()
{
    uint cmdIndex = gl_GlobalInvocationID.x;
    if (cmdIndex >= params.command_count)
        return;

    DrawElementsIndirectCommand cmd = draw_cmd_out[cmdIndex];

    if (cmd.primCount == 0)
        return;

    int batch = draw_to_batch[cmdIndex];

    // atomic counter per batch
    uint localIndex = atomicAdd(counts[batch], 1);

    uint writeIndex = params.command_count +
                      multidraw_batches[batch * 2] +
                      localIndex;

    draw_cmd_out[writeIndex] = cmd;
}
