#include "ShaderBufferShared.txt"

out vec4 FS_OUT_Color;

in vec4 FS_IN_Color;

#ifdef USE_SOLID_COLOR
layout(std140, binding = 16) uniform MbSimpleFragPushConstsUbo { MbSimpleFragPushConsts pcf; };
#endif

void main()
{
#ifdef USE_SOLID_COLOR
	FS_OUT_Color = pow(pcf.solid_color,vec4(2.2));
#else
	FS_OUT_Color = pow(vec4(FS_IN_Color.rgb,1.0),vec4(2.2));
#endif
}
