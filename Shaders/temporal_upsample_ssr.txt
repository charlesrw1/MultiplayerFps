

layout(location = 0) in vec2 texCoord;
layout(location = 0) out vec4 FS_OUT_Color;

#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"
layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};
layout (binding = 7, std140) uniform SsrParamsUbo { SsrParams params; };

layout(binding=0) uniform sampler2D resolvedSSR;
layout(binding=1) uniform sampler2D prevTemporalSSR;
layout(binding=2) uniform sampler2D sceneMotion;

float Luminance(vec3 c) { return dot(c, vec3(0.2126, 0.7152, 0.0722)); }

vec4 ClipToAABB(vec4 color, vec4 aabbMin, vec4 aabbMax)
{
	vec4 center = 0.5 * (aabbMax + aabbMin);
	vec4 extents = 0.5 * (aabbMax - aabbMin) + 1e-5;
	vec4 offset = color - center;
	vec4 ts = abs(extents / (offset + 1e-10));
	float t = min(min(ts.x, ts.y), min(ts.z, ts.w));
	t = clamp(t, 0.0, 1.0);
	return center + offset * t;
}

void main()
{
	vec2 uv = texCoord;

	// Motion vectors: 0.5 = no motion, offset in UV space
	vec2 velocity_uv = textureLod(sceneMotion, uv, 0.0).rg;
	vec2 velocityDelta = velocity_uv - vec2(0.5);
	float velocityLength = length(velocityDelta);

	// Reproject to previous frame
	vec2 ndc = uv * 2.0 - 1.0;
	vec2 velNDC = velocityDelta * 2.0;
	vec2 prevUV = clamp((ndc + velNDC) * 0.5 + 0.5, vec2(0.0), vec2(1.0));

	vec4 current = textureLod(resolvedSSR, uv, 0.0);
	vec2 du = vec2(params.ssrTexelSize.x, 0.0);
	vec2 dv = vec2(0.0, params.ssrTexelSize.y);

	// 3x3 neighborhood for AABB clamping
	vec4 tl = textureLod(resolvedSSR, uv - dv - du, 0.0);
	vec4 tc = textureLod(resolvedSSR, uv - dv, 0.0);
	vec4 tr = textureLod(resolvedSSR, uv - dv + du, 0.0);
	vec4 ml = textureLod(resolvedSSR, uv - du, 0.0);
	vec4 mc = current;
	vec4 mr = textureLod(resolvedSSR, uv + du, 0.0);
	vec4 bl = textureLod(resolvedSSR, uv + dv - du, 0.0);
	vec4 bc = textureLod(resolvedSSR, uv + dv, 0.0);
	vec4 br = textureLod(resolvedSSR, uv + dv + du, 0.0);

	vec4 currentMin = min(tl, min(tc, min(tr, min(ml, min(mc, min(mr, min(bl, min(bc, br))))))));
	vec4 currentMax = max(tl, max(tc, max(tr, max(ml, max(mc, max(mr, max(bl, max(bc, br))))))));
	vec4 currentAvg = (tl + tc + tr + ml + mc + mr + bl + bc + br) / 9.0;

	// Sharpening
	current += (current - currentAvg) * 0.1;

	// Clamp history to neighborhood AABB with velocity-adaptive margin
	float lumaOffset = abs(Luminance(currentAvg.rgb) - Luminance(current.rgb));
	float aabbMargin = mix(4.0, 0.25, clamp(velocityLength * 100.0, 0.0, 1.0)) * lumaOffset;
	vec4 previous = textureLod(prevTemporalSSR, prevUV, 0.0);
	previous = ClipToAABB(previous, currentMin - aabbMargin, currentMax + aabbMargin);

	// Velocity-adaptive temporal blend
	float response = params.temporalResponse * (1.0 - velocityLength * 8.0);
	current = mix(current, previous, clamp(response, 0.0, 1.0));
	current = clamp(current, vec4(0.0), vec4(65000.0));

	FS_OUT_Color = current;
}
