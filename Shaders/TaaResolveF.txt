layout(location = 0) in vec2 texCoord;
layout(location = 0) out vec3 FS_OUT_Color;
#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"
layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};
layout (binding = 7, std140) uniform TemporalParamsUbo { TemporalParams params; };

layout(binding=0) uniform sampler2D sceneColor;
layout(binding=1) uniform sampler2D lastColor;
layout(binding=2) uniform sampler2D sceneDepth;
layout(binding=3) uniform sampler2D sceneMotion;
layout(binding=4) uniform sampler2D lastSceneMotion;

// [Pedersen, 2016, "Temporal Reprojection Anti-Aliasing in INSIDE"]
vec4 ClipToAABB(vec4 color, vec4 minimum, vec4 maximum)
{
	vec4 center = (maximum + minimum) * 0.5;
	vec4 extents = (maximum - minimum) * 0.5;
	vec4 shift = color - center;
	vec4 absUnit = abs(shift / max(extents, vec4(0.0001)));
	float maxUnit = max(max(absUnit.x, absUnit.y), absUnit.z);
	return maxUnit > 1.0 ? center + (shift / maxUnit) : color;
}

// [Jimenez, 2016, "Filmic SMAA", p92 — Catmull-Rom 5-tap]
vec4 SampleTextureCatmullRom5(sampler2D tex, vec2 uv)
{
	vec2 texsiz = vec2(textureSize(tex, 0).xy);
	vec4 rtMetrics = vec4(1.0 / texsiz.xy, texsiz.xy);

	vec2 position = rtMetrics.zw * uv;
	vec2 centerPosition = floor(position - 0.5) + 0.5;
	vec2 f = position - centerPosition;
	vec2 f2 = f * f;
	vec2 f3 = f * f2;

	const float c = 0.5;
	vec2 w0 =        -c  * f3 +  2.0 * c         * f2 - c * f;
	vec2 w1 =  (2.0 - c) * f3 - (3.0 - c)        * f2         + 1.0;
	vec2 w2 = -(2.0 - c) * f3 + (3.0 -  2.0 * c) * f2 + c * f;
	vec2 w3 =         c  * f3 -                c * f2;

	vec2 w12 = w1 + w2;
	vec2 tc12 = rtMetrics.xy * (centerPosition + w2 / w12);
	vec3 centerColor = textureLod(tex, vec2(tc12.x, tc12.y), 0.0).rgb;

	vec2 tc0 = rtMetrics.xy * (centerPosition - 1.0);
	vec2 tc3 = rtMetrics.xy * (centerPosition + 2.0);
	vec4 color = vec4(textureLod(tex, vec2(tc12.x, tc0.y ), 0.0).rgb, 1.0) * (w12.x * w0.y ) +
	             vec4(textureLod(tex, vec2(tc0.x,  tc12.y), 0.0).rgb, 1.0) * (w0.x  * w12.y) +
	             vec4(centerColor,                                     1.0) * (w12.x * w12.y) +
	             vec4(textureLod(tex, vec2(tc3.x,  tc12.y), 0.0).rgb, 1.0) * (w3.x  * w12.y) +
	             vec4(textureLod(tex, vec2(tc12.x, tc3.y ), 0.0).rgb, 1.0) * (w12.x * w3.y );
	return vec4(color.rgb / color.a, 1.0);
}

float Luminance(vec3 c)
{
	return dot(c, vec3(0.299, 0.587, 0.114));
}

vec2 GetDilatedVelocity(vec2 uv, vec2 invsize)
{
	float closestDepth = 100.0;
	vec2 closestUVOffset = vec2(0.0);
	for(int j = -1; j <= 1; ++j)
	{
		for(int i = -1; i <= 1; ++i)
		{
			vec2 uvOffset = vec2(i, j) * invsize;
			float neighborDepth = textureLod(sceneDepth, uv + uvOffset, 0.0).r;
			if(neighborDepth < closestDepth)
			{
				closestUVOffset = uvOffset;
				closestDepth = neighborDepth;
			}
		}
	}
	return textureLod(sceneMotion, uv + closestUVOffset, 0.0).rg;
}

void main()
{
	vec2 invsize = 1.0 / g.viewport_size.xy;
	vec2 input_uv = gl_FragCoord.xy * invsize;

	// Velocity dilation: use closest-depth neighbor's velocity for clean edges
	vec2 velocity_uv = GetDilatedVelocity(input_uv, invsize);

	// Reproject to previous frame
	vec2 velocity_ndc = velocity_uv * 2.0 - 1.0;
	float velocityLength = length(velocity_ndc);
	vec2 prevUV = (input_uv * 2.0 - 1.0 + velocity_ndc) * 0.5 + 0.5;
	prevUV = clamp(prevUV, vec2(0.0), vec2(1.0));

	const float HDR_CLAMP_MAX = 65000.0;

	// Build 3x3 neighborhood min/max and average
	vec3 current = min(texture(sceneColor, input_uv).rgb, vec3(HDR_CLAMP_MAX));
	vec4 neighborhoodMin = vec4(9999.0);
	vec4 neighborhoodMax = vec4(-9999.0);
	vec4 neighborhoodSum = vec4(0.0);
	for(int x = -1; x <= 1; ++x)
	{
		for(int y = -1; y <= 1; ++y)
		{
			vec4 neighbor = vec4(min(texture(sceneColor, input_uv + vec2(x, y) * invsize).rgb, vec3(HDR_CLAMP_MAX)), 1.0);
			neighborhoodMin = min(neighborhoodMin, neighbor);
			neighborhoodMax = max(neighborhoodMax, neighbor);
			neighborhoodSum += neighbor;
		}
	}
	vec4 neighborhoodAvg = neighborhoodSum / 9.0;
	vec4 currentVec = vec4(current, 1.0);

	// Sample history: Catmull-Rom is sharper but rings at silhouettes, bilinear is safer
	vec3 historyRaw = (params.use_catmull_rom != 0)
		? SampleTextureCatmullRom5(lastColor, prevUV).rgb
		: textureLod(lastColor, prevUV, 0.0).rgb;
	vec4 history = vec4(clamp(historyRaw, vec3(0.0), vec3(HDR_CLAMP_MAX)), 1.0);

	// Clip history to neighborhood AABB with luma-adaptive margin
	float lumaOffset = abs(Luminance(neighborhoodAvg.rgb) - Luminance(currentVec.rgb));
	float aabbMargin = mix(4.0, 0.25, clamp(velocityLength * 100.0, 0.0, 1.0)) * lumaOffset;
	history = ClipToAABB(history, neighborhoodMin - aabbMargin, neighborhoodMax + aabbMargin);

	// Velocity-adaptive blend factor (Flax-style)
	float motion = clamp(velocityLength * 1000.0, 0.0, 1.0);
	float blendfactor = mix(params.stationary_blending, params.motion_blending, motion);

	// Luminance weighting for flicker reduction
	float lumaCurrent = Luminance(currentVec.rgb);
	float lumaHistory = Luminance(history.rgb);
	float currentWeight = (1.0 - blendfactor) / (1.0 + lumaCurrent);
	float historyWeight = blendfactor / (1.0 + lumaHistory);
	vec3 color = (currentVec.rgb * currentWeight + history.rgb * historyWeight) /
	             max(currentWeight + historyWeight, 0.00001);

	// Sharpening: applied after blend to counteract temporal blur
	color += (color - neighborhoodAvg.rgb) * params.sharpness;
	color = clamp(color, vec3(0.0), vec3(HDR_CLAMP_MAX));

	// Velocity disocclusion (elopezr "length of difference" metric)
	vec2 lastVelocityUv = textureLod(lastSceneMotion, prevUV, 0.0).rg;
	float vel_len = length(lastVelocityUv - velocity_uv);
	float velocityDisocclusion = clamp(
		pow(max((vel_len - params.doc_vel_bias) * params.doc_mult - params.doc_bias, 0.0), params.doc_pow),
		0.0, 1.0);

	// Off-screen rejection
	float offscreen = any(greaterThanEqual(abs(prevUV * 2.0 - 1.0), vec2(1.0))) ? 1.0 : 0.0;

	// Blend towards neighborhood on disocclusion or off-screen
	vec3 neighborhoodSharp = mix(neighborhoodAvg.rgb, current, 0.5);
	color = mix(color, neighborhoodSharp, clamp(velocityDisocclusion + offscreen, 0.0, 1.0));

	FS_OUT_Color = clamp(color, vec3(0.0), vec3(HDR_CLAMP_MAX));
}
