float AABBIntersect(vec3 bmin,vec3 bmax, Ray r)
{
	vec3 tmin =(bmin-r.pos)*r.inv_dir;
	vec3 tmax =(bmax-r.pos)*r.inv_dir;
	vec3 t1 =min(tmin,tmax);
	vec3 t2 = max(tmin,tmax);
	float tnear = max(max(t1.x,t1.y),t1.z);
	float tfar = min(min(t2.x,t2.y),t2.z);
	
	return (tnear>tfar||tfar<0) ? -INF : tnear;
}

float RayTriIntersect2(vec3 ro, vec3 rd, vec3 v0, vec3 v1, vec3 v2, out vec3 bary)
{
	vec4 uvt;
	const vec3 edge1 = v1 - v0;
	const vec3 edge2 = v2 - v0;
	const vec3 h = cross(rd, edge2);
	const float a = dot(edge1, h);
//	if (a > -0.0000001 && a < 0.0000001) return -1; 
	const float f = 1 / a;
	const vec3 s = ro - v0;
	uvt.x = f * dot(s, h);
	//if (u < 0 || u > 1) return -1;
	const vec3 q = cross(s, edge1);
	uvt.y = f * dot(rd, q);
	//if (v < 0 || u + v > 1) return -1;
	uvt.z = f * dot(edge2, q);
	uvt.w = 1-uvt.x-uvt.y;
	if(all(greaterThanEqual(uvt,vec4(0.0))))
	{
		bary = uvt.xyw;
		return uvt.z;
	}
	return -1;
}

struct TraceOut
{
	float t;
	vec3 hit_normal;
	int mat_hit;
};

TraceOut trace_ray(vec3 ro, vec3 rd)
{
	Ray out_ray;
	out_ray.pos = ro;
	out_ray.dir = rd;
	out_ray.inv_dir = 1.f/(rd);
	
	uint stack[64];
	stack[0]=0;
	
	
	uint node;
	int stack_count = 1;

	float t = INF;
	//vec3 bary_near;
	int idx=-1;
	//int tris_hit = 0;
	int mat_hit = 0;
	vec3 closest_normal;
	while (stack_count > 0)
	{
		node=stack[--stack_count];
		
		
		// Fetch data
		int node_leftnode = Nodes.data[node].left_node;
		int node_count = Nodes.data[node].count;

		if (node_count != -1)
		{
			int index_count = node_count;
			int index_start = node_leftnode;
			for (int i = 0; i < index_count; i++) {
				
				int mesh_element_index = References.data[index_start+i]*3;
				uint i0 = Indicies.data[mesh_element_index];
				uint i1 = Indicies.data[mesh_element_index+1];
				uint i2 = Indicies.data[mesh_element_index+2];
				vec3 bary;
				vec3 v0 = Verticies.data[i0].xyz;
				vec3 v1 = Verticies.data[i1].xyz;
				vec3 v2 = Verticies.data[i2].xyz;
				
				
				float t_hit = RayTriIntersect2(
					ro,rd,
					v0,v1,v2,
					bary);		
				if(t_hit>0.00001 && t_hit < t) {
					t = t_hit;
					idx = mesh_element_index;
					//bary_near=bary;
					mat_hit = int(Verticies.data[i0].w);
					closest_normal = normalize(cross(v1-v0,v2-v0));
				}
			}
			continue;
		}

		float left_dist, right_dist;
		left_dist = AABBIntersect(Nodes.data[node_leftnode].bmin.xyz,Nodes.data[node_leftnode].bmax.xyz,out_ray);
		right_dist = AABBIntersect(Nodes.data[node_leftnode+1].bmin.xyz,Nodes.data[node_leftnode+1].bmax.xyz,out_ray);
		
		bool left_bool = left_dist>-INF&&left_dist<t;
		bool right_bool = right_dist>-INF&&right_dist<t;

		if (left_bool && right_bool) {
			bool push_right_first = left_dist<right_dist;
			stack[stack_count++]=node_leftnode+int(push_right_first);
			stack[stack_count++]=node_leftnode+int(!push_right_first);
		
		
			//if (left_dist < right_dist) {
			//	stack[stack_count++]=node_leftnode+1;
			//	stack[stack_count++]=node_leftnode;
			//}
			//else {
			//
			//	stack[stack_count++]=node_leftnode;
			//	stack[stack_count++]=node_leftnode+1;
			//}
		}
		else if (left_bool) {
			stack[stack_count++]=node_leftnode;
		}
		else if (right_bool) {
			stack[stack_count++]=node_leftnode+1;
		}
	}
	
	if(idx==-1) {
		t=-1;
	}

	TraceOut outhit;
	outhit.t = t;
	outhit.hit_normal = closest_normal;
	outhit.mat_hit = mat_hit;
	return outhit;
}

vec3 RTXGISphericalFibonacci(float sampleIndex, float numSamples)
{
    const float b = (sqrt(5.0) * 0.5 + 0.5) - 1.0;
    float phi = 3.14159*2.0 * fract(sampleIndex * b);
    float cosTheta = 1.0 - (2.0 * sampleIndex + 1.0) * (1.0 / numSamples);
    float sinTheta = sqrt(clamp(1.f - (cosTheta * cosTheta),0.0,1.0));

    return vec3((cos(phi) * sinTheta), (sin(phi) * sinTheta), cosTheta);
}


vec3 get_probe_world_from_coords(ivec3 coords)
{
return  volume_origin+vec3(coords)*volume_spacing;
}

vec2 OctahedralEncode(vec3 normal_in)
{
	vec3 normal = normal_in.xzy;

	normal = normalize(normal);
	float l1norm = abs(normal.x) + abs(normal.y) + abs(normal.z);
	vec2 result = normal.xy * (1.0 / l1norm);
	
	if (normal.z < 0.0) {
		vec2 oldResult = result;
		result.x = (1.0 - abs(oldResult.y)) * (oldResult.x >= 0.0 ? 1.0 : -1.0);
		result.y = (1.0 - abs(oldResult.x)) * (oldResult.y >= 0.0 ? 1.0 : -1.0);
	}
	
	return result * 0.5 + 0.5; // Map to [0,1]
}

vec2 get_depth_probe_uv(vec2 octant_uv, ivec3 probe_coord) {
	const int DEPTH_IN_SIZE = 14;
	const int DEPTH_TILE_SIZE = 16;
	
	vec2 coords;
	coords.x = probe_coord.x*DEPTH_TILE_SIZE + (vol_grid.x*probe_coord.y*DEPTH_TILE_SIZE);
	coords.y = probe_coord.z*DEPTH_TILE_SIZE;
	
	coords += vec2(octant_uv*vec2(DEPTH_IN_SIZE))+vec2(1);
	vec2 final_uv = vec2(coords)/vec2(vol_grid.x*vol_grid.y*DEPTH_TILE_SIZE,vol_grid.z*DEPTH_TILE_SIZE);

	return final_uv;
}

vec2 get_irrad_probe_uv(vec2 octant_uv, ivec3 probe_coord) {
	const int DEPTH_IN_SIZE = 6;
	const int DEPTH_TILE_SIZE = 8;
	
	vec2 coords;
	coords.x = probe_coord.x*DEPTH_TILE_SIZE + (vol_grid.x*probe_coord.y*DEPTH_TILE_SIZE);
	coords.y = probe_coord.z*DEPTH_TILE_SIZE;
	
	coords += vec2(octant_uv*vec2(DEPTH_IN_SIZE))+vec2(1);
	vec2 final_uv = vec2(coords)/vec2(vol_grid.x*vol_grid.y*DEPTH_TILE_SIZE,vol_grid.z*DEPTH_TILE_SIZE);

	return final_uv;
}


vec3 calculate_irradiance(vec3 direction, vec3 worldspace)
{
	
		
	vec3 biased_world_position = worldspace+direction*0.1;
	
	
	
	ivec3 base_probe_coords = ivec3((biased_world_position-volume_origin)/volume_spacing);
	base_probe_coords = clamp(base_probe_coords,ivec3(0), 
	ivec3(vol_grid)-ivec3(1)
	);
	vec3 base_probe_pos = get_probe_world_from_coords(base_probe_coords);
	
	vec3 alpha = clamp(
		(biased_world_position-base_probe_pos)/volume_spacing,
		vec3(0),
		vec3(1));
	

	vec3 irradiance=vec3(0.0);
	float accum_weight = 0.0;
	float cheb_weight_total = 0.0;
	
	for(int probeIndex = 0;probeIndex < 8; probeIndex++) {
		ivec3 adjacent_probe_ofs = ivec3(probeIndex, probeIndex>>1, probeIndex>>2)&ivec3(1);
		ivec3 adjacent_probe_coords = clamp(
			base_probe_coords+adjacent_probe_ofs,
			ivec3(0),
			ivec3(vol_grid)-ivec3(1));
		vec3 adj_probe_worldpos = get_probe_world_from_coords(adjacent_probe_coords);
		

		vec3 biased_pos_to_adj_probe = normalize(adj_probe_worldpos-biased_world_position);
		float biased_pos_to_adj_probe_len = length(adj_probe_worldpos-biased_world_position);
		
		vec3 trilinear = max(vec3(0.001), 
			mix(vec3(1.0-alpha),vec3(alpha),vec3(adjacent_probe_ofs)));
		float trilinear_weight = trilinear.x*trilinear.y*trilinear.z;
		float weight = 1.0;
		float warp_shading = dot(biased_pos_to_adj_probe,direction)*0.5 + 0.5;
		weight *= (warp_shading*warp_shading)+0.2;
		
		vec2 octant_coords = OctahedralEncode(-biased_pos_to_adj_probe);
		
		vec2 probe_texture_uv = get_depth_probe_uv(
		octant_coords,
		adjacent_probe_coords);
		
		vec2 filtered_distance = vec2(2.0)*texture(probe_depth, probe_texture_uv).rg;
		
		float variance = abs(
		(filtered_distance.x*filtered_distance.x) - filtered_distance.y);
		
		float chebyshev_weight = 1.0;
		if(biased_pos_to_adj_probe_len > filtered_distance.x) {
			float v = biased_pos_to_adj_probe_len - filtered_distance.x;
			chebyshev_weight = variance/(variance+(v*v));
			chebyshev_weight = max(
			chebyshev_weight*chebyshev_weight*chebyshev_weight,
			0.0
			);
		}
		
		weight *= max(0.05, chebyshev_weight);
		weight = max(0.000001, weight);
		const float crush_thresh = 0.2;
		if(weight<crush_thresh) {
			weight *= (weight*weight)*(1.0/(crush_thresh*crush_thresh));
		}
		weight *= trilinear_weight;
		
		octant_coords = OctahedralEncode(direction);
		probe_texture_uv = get_irrad_probe_uv(octant_coords,adjacent_probe_coords);
		
		vec3 probe_irrad = texture(probe_irradiance, probe_texture_uv).rgb;
		//probe_irrad = pow(probe_irrad, vec3(2.5));
		
		irradiance += (weight*probe_irrad);
		accum_weight += weight;
		// holy crap so much code!
		
		//cheb_weight_total+= chebyshev_weight;
	}
	
	irradiance *= (1.0/accum_weight);
	
	return irradiance;

}
uniform bool do_irrad_calcs = true;

vec3 evaluate_lighting(vec3 pos, vec3 normal, vec3 albedo)
{
	vec3 irrad = vec3(0.0);
	if(do_irrad_calcs) {
		irrad =  calculate_irradiance(normal,pos)*albedo;
		
	}
	//vec3 sun_dir = normalize(vec3(1,-1,1));
	//TraceOut outsun = trace_ray(pos+normal*0.01, -sun_dir);
	//if(outsun.t > 1000) {
	//	irrad += albedo*1.0;
	//}
	return irrad;
}


void main()
{
	//uvec3 global_id = gl_GlobalInvocationID.z;
	
	// Determine probe and ray index
	int probe_number = int(gl_GlobalInvocationID.x);
	
	if(probe_number > vol_grid.x*vol_grid.y*vol_grid.z)
		return;
	
		// Get probe position in grid
	int probe_x = probe_number % int(vol_grid.x);
	int probe_y = (probe_number / int(vol_grid.x)) % int(vol_grid.y);
	int probe_z = probe_number / (int(vol_grid.x) * int(vol_grid.y));
	
	
	vec3 probe_pos = volume_origin.xyz + vec3(probe_x, probe_y, probe_z) * vec3(volume_spacing);
	
	for(int ray_idx = 0; ray_idx < NUM_RAYS; ray_idx += 1) {
		// Generate ray direction from ray index (spherical distribution)

		vec3 ray_dir = RTXGISphericalFibonacci(ray_idx,NUM_RAYS);
		
	
		// Trace ray from probe
		TraceOut hit_result = trace_ray(probe_pos, ray_dir);
		//vec3 hit_albedo = hit_color.xyz;
		
		vec3 incoming_radiance = vec3(0.0);
		float out_distance = 1000.0;
		if(hit_result.t < 0 || hit_result.t > 1000) {
			// eval sky
			//incoming_radiance = vec3(1);
			incoming_radiance = textureLod(Skylight,ray_dir, 2.0).rgb;
		}
		else {
			out_distance = hit_result.t;
			incoming_radiance = evaluate_lighting(
			probe_pos+ray_dir*hit_result.t,
			hit_result.hit_normal, 
			Albedos.data[hit_result.mat_hit].xyz
			);
		}
		
		// have lighting, have depth
	
		// Write to hit buffer
		int buffer_idx = probe_number * NUM_RAYS + ray_idx;
		hitbuffer[buffer_idx].dir_dist = vec4(ray_dir,out_distance*sign(-dot(ray_dir,hit_result.hit_normal)));
		hitbuffer[buffer_idx].light = vec4(incoming_radiance, 1.0);
	}
}
