vec3 RTXGISphericalFibonacci(float sampleIndex, float numSamples)
{
    const float b = (sqrt(5.0) * 0.5 + 0.5) - 1.0;
    float phi = 3.14159*2.0 * fract(sampleIndex * b);
    float cosTheta = 1.0 - (2.0 * sampleIndex + 1.0) * (1.0 / numSamples);
    float sinTheta = sqrt(clamp(1.f - (cosTheta * cosTheta),0.0,1.0));


	phi += ray_sample_randomness;

    return vec3((cos(phi) * sinTheta), (sin(phi) * sinTheta), cosTheta);
}

vec3 evaluate_punctual_lights(vec3 worldspace, vec3 albedo, vec3 normal_t) {
vec3 output_light_accum = vec3(0.0);
	for(int light_index=0;light_index<num_lights;light_index++) {

	
		vec4 _position_radius = light_objects[light_index].position_radius;
		vec3 position = _position_radius.xyz;
		float radius = _position_radius.w;
		vec3 color = light_objects[light_index].light_color.xyz;
		bool is_spot_light = bool(light_objects[light_index].flags&1);
		vec3 spot_normal =  light_objects[light_index].spot_normal.xyz;
		float spot_angle = light_objects[light_index].spot_angle;
		float spot_inner = light_objects[light_index].spot_inner;


		
		float distance_to_light = length(position-worldspace);
		if(distance_to_light>radius)
			continue;
		
		vec3 L = (position - worldspace)/distance_to_light;
		
		float distance_to_light_2 = distance_to_light*distance_to_light;
		float radius_2 = radius*radius;
		float attenuation = 1.0/distance_to_light_2 * clamp(1-distance_to_light_2/radius_2,0.0,1.0);
		
		vec3 out_color = pbr_calculate_punctual(color, L, worldspace, albedo.xyz, 1, 0, normal_t.xyz);
		
		if(is_spot_light) {
			float angle = dot(-L, spot_normal);
		#ifndef COOKIE
			attenuation *= clamp((-spot_angle+angle) / (spot_inner-spot_angle), 0.0, 1.0);    
		#endif
			
		}
		
		output_light_accum += out_color * attenuation;
	}
	return output_light_accum;
	}


vec3 evaluate_lighting(vec3 pos, vec3 normal, vec3 albedo)
{
	vec3 irrad = vec3(0.0);
	if(do_irrad_calcs) {
		vec3 biased_world_position = pos+normal*0.1;
	
		irrad =  calculate_irradiance(normal,pos,biased_world_position);
		
	}
	vec3 sun_dir = normalize(vec3(1,-1,1));
	TraceOut outsun = trace_ray(pos, -sun_dir);
	if(outsun.t < 0) {
		irrad += max(dot(-sun_dir,normal),0.0)*0.0;
	}
	
	irrad = irrad * albedo / 3.14159;
	irrad += evaluate_punctual_lights(pos,albedo,normal);
	
	return irrad;
}


void main()
{
	const int global_index = int(gl_GlobalInvocationID.x);
	const int volume_index = get_volume_index_for_global_index(global_index);
	const int probe_number = global_index-volumes[volume_index].size_offset.w;
	const ivec3 vol_grid = volumes[volume_index].size_offset.xyz;
	const vec3 volume_spacing = volumes[volume_index].density.xyz;
	const vec3 volume_origin = volumes[volume_index].origin_pri.xyz;
	
	
	if(probe_number > vol_grid.x*vol_grid.y*vol_grid.z)
		return;
	
		// Get probe position in grid
	int probe_x = probe_number % int(vol_grid.x);
	int probe_y = (probe_number / int(vol_grid.x)) % int(vol_grid.y);
	int probe_z = probe_number / (int(vol_grid.x) * int(vol_grid.y));
	
	
	vec3 probe_pos = get_probe_world_from_coords(ivec3(probe_x,probe_y,probe_z), volume_origin,volume_spacing);
//	volume_origin.xyz + vec3(probe_x, probe_y, probe_z) * vec3(volume_spacing);
	
	for(int ray_idx = 0; ray_idx < NUM_RAYS; ray_idx += 1) {
		// Generate ray direction from ray index (spherical distribution)

		vec3 ray_dir = RTXGISphericalFibonacci(ray_idx,NUM_RAYS);
		
	
		// Trace ray from probe
		TraceOut hit_result = trace_ray(probe_pos, ray_dir);
		//vec3 hit_albedo = hit_color.xyz;
		
		vec3 incoming_radiance = vec3(0.0);
		float out_distance = 1000.0;
		if(hit_result.t < 0 || hit_result.t > 1000) {
			// eval sky
			//incoming_radiance = vec3(1);
			incoming_radiance = textureLod(Skylight,ray_dir, 2.0).rgb;
		}
		else {
			vec3 pos_hit = probe_pos+ray_dir*hit_result.t;
			vec3 pos_to_eval_from = pos_hit - ray_dir*0.03 + hit_result.hit_normal * 0.03;
			out_distance = hit_result.t;
			incoming_radiance = evaluate_lighting(pos_to_eval_from,hit_result.hit_normal, Albedos.data[hit_result.mat_hit].xyz);
		}
		
		// have lighting, have depth
	
		// Write to hit buffer
		int buffer_idx = global_index * NUM_RAYS + ray_idx;
		hitbuffer[buffer_idx].dir_dist = vec4(ray_dir,out_distance*sign(-dot(ray_dir,hit_result.hit_normal)));
		hitbuffer[buffer_idx].light = vec4(incoming_radiance, 1.0);
	}
}
