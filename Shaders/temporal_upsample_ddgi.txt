


layout(location = 0) out vec4 FS_OUT_Color;

#include "SharedGpuTypes.txt"
#include "ShaderBufferShared.txt"
layout (binding = 0, std140) uniform Ubo_View_Constant_Buffer {
	Ubo_View_Constants_Struct g;
};
layout (binding = 7, std140) uniform TemporalParamsUbo { TemporalParams params; };

layout(binding=0) uniform sampler2D halfResColor;
layout(binding=1) uniform sampler2D lastFullColor;
layout(binding=2) uniform sampler2D sceneDepth;
layout(binding=3) uniform sampler2D sceneMotion;
layout(binding=4) uniform sampler2D lastSceneMotion;


vec2 reproject_to_last(vec2 uv)
{
	float depth = textureLod(sceneDepth,uv,0.0).r;
	uv = uv*2.0 - vec2(1.0);
	vec4 worldspace = g.inv_viewproj * vec4(uv.x,uv.y,depth,1.0);
	worldspace /= worldspace.w;
	vec4 lastClipspace = params.lastViewProj * vec4(worldspace.xyz,1.0);
	lastClipspace /= lastClipspace.w;
	
	vec2 reproject_uv = lastClipspace.xy*0.5 + vec2(0.5);
	return reproject_uv;
}
// to save my fucking eyeballs
float sanitize_float(float v)
{
    if (isnan(v) || isinf(v))
        return 0.0;
    return v;
}
vec3 sanitize(vec3 v)
{
    return vec3(
        sanitize_float(v.x),
        sanitize_float(v.y),
		sanitize_float(v.z)
    );
}

float linear_depth(float d)
{
	return g.near / d;
}

// Clips q toward the aabb center p_clip so the intersection point with the
// box lies on the p_clip->q ray (Playdead's TAA "clip_aabb") -- avoids the
// hue-shift that a plain per-channel clamp(q,min,max) introduces.
vec3 clip_aabb(vec3 aabb_min, vec3 aabb_max, vec3 p, vec3 q)
{
	vec3 p_clip = 0.5 * (aabb_max + aabb_min);
	vec3 e_clip = 0.5 * (aabb_max - aabb_min) + vec3(1e-7);
	vec3 v_clip = q - p_clip;
	vec3 v_unit = v_clip / e_clip;
	vec3 a_unit = abs(v_unit);
	float ma_unit = max(a_unit.x, max(a_unit.y, a_unit.z));
	return (ma_unit > 1.0) ? (p_clip + v_clip / ma_unit) : q;
}

// Depth-aware bilinear upsample of the half-res buffer: rejects taps whose
// source pixel is on the wrong side of a depth discontinuity (avoids light
// leaking across edges), and falls back to the closest-depth tap when all
// 4 taps disagree strongly (e.g. thin foreground object over background).
vec3 bilateral_upsample(vec2 input_uv, float full_linear_depth, vec2 halfres_invsize, vec2 halfres_size)
{
	const vec2 half_pos = input_uv * halfres_size - vec2(0.5);
	const vec2 half_base = floor(half_pos);
	const vec2 frac = half_pos - half_base;

	vec3 sum = vec3(0.0);
	float wsum = 0.0;
	vec3 closest_color = vec3(0.0);
	float closest_depth_diff = 999999.0;

	for (int y = 0; y <= 1; y++) {
		for (int x = 0; x <= 1; x++) {
			const vec2 tap = half_base + vec2(x,y);
			const vec2 tap_uv = (tap + vec2(0.5)) * halfres_invsize;

			const float tap_depth = linear_depth(textureLod(sceneDepth, tap_uv, 0.0).r);
			const float depth_diff = abs(tap_depth - full_linear_depth);
			const float depth_w = 1.0 / (1.0 + depth_diff*depth_diff * 0.25);

			const float bw = (x==0 ? (1.0-frac.x) : frac.x) * (y==0 ? (1.0-frac.y) : frac.y);
			const vec3 tap_color = textureLod(halfResColor, tap_uv, 0.0).rgb;

			const float w = bw*depth_w;
			sum += tap_color*w;
			wsum += w;

			if (depth_diff < closest_depth_diff) {
				closest_depth_diff = depth_diff;
				closest_color = tap_color;
			}
		}
	}

	return (wsum > 0.0001) ? (sum/wsum) : closest_color;
}

vec3 temporal_upscale()
{
	const ivec2 input_texel = ivec2(gl_FragCoord.xy);
	const vec2 invsize = 1.0 / g.viewport_size.xy;
	const vec2 input_uv = vec2(vec2(input_texel)+vec2(0.5)) * invsize;
	const vec2 halfres_size = g.viewport_size.xy * 0.5;
	const vec2 halfres_invsize = invsize * 2.0;

	const ivec2 my_texel_offset = ivec2(uvec2(input_texel)%uvec2(2,2));
	const float weight = (my_texel_offset == params.halfres_texel_offset)? 1.0 : 0.00;

	const float full_linear_depth = linear_depth(texelFetch(sceneDepth,input_texel,0).r);
	const vec3 current = bilateral_upsample(input_uv, full_linear_depth, halfres_invsize, halfres_size);


	vec2 velocity_uv = texelFetch(sceneMotion,input_texel,0).rg;
	//if(dilate_velocity)
	//	velocity_uv = GetVelocity(input_uv);
	
	if(length(velocity_uv)<0.00001)
		velocity_uv=vec2(0.5);
	
	//velocity_uv = vec2(0.5);
	vec2 ndc_input = input_uv*2.0-vec2(1.0);
	vec2 velocity_uv_ndc = velocity_uv*2.0-vec2(1.0);
	ndc_input += velocity_uv_ndc;
	vec2 uv_to_use = ndc_input * 0.5 + 0.5;
	//uv_to_use = reproject_to_last(input_uv);
	uv_to_use = clamp(uv_to_use,vec2(0.0),vec2(1.0));
	
	vec2 actual_uv = reproject_to_last(input_uv);
	//uv_to_use = actual_uv;
	if(params.use_reproject != 0)
		uv_to_use = actual_uv;
	
	//vec3 last = SampleTextureCatmullRom5(lastColor,uv_to_use).rgb;
	vec3 last = textureLod(lastFullColor,uv_to_use,0).rgb;// looks better than the cubic filter...
	vec2 lastVelocityUv = textureLod(lastSceneMotion,uv_to_use,0).rg;
	float vel_len = length(lastVelocityUv-velocity_uv);
	// max(...,0.0): base must not go negative, else pow() with a non-integer
	// exponent yields NaN (was silently eaten by sanitize() as a black pixel).
	float velocityDisocclusion = clamp(pow(max((vel_len-params.doc_vel_bias)*params.doc_mult-params.doc_bias,0.0),params.doc_pow),0.0,1.0);

	// when dioccluded, lerp to 1.0
	// else, lerp to weight. weight is 0 if this texel wasnt fetched this frame, else 1. so ddgi is basically updating at 15hz (assuming 60hz rendering)
	const float weight_with_dissocclusion = mix(weight,1.0,velocityDisocclusion);

	// variance clipping in *half-res* texel space (steps by halfres_invsize,
	// not fullres invsize, so the box actually spans the neighborhood).
	// Plain min/max AABB collapses to ~current on the low-variance smooth
	// gradients typical of diffuse GI, which makes params.amt look like it
	// does nothing (history always gets clamped back to current regardless
	// of blend weight). Fitting an ellipsoid to the neighborhood mean/stddev
	// instead leaves room to blend on smooth regions while still bounding
	// history hard enough near real discontinuities to fight ghosting.
	vec3 colorSum = vec3(0.0);
	vec3 colorSqSum = vec3(0.0);
	for(int x = -1;x<=1;x++) {
		for(int y=-1;y<=1;y++) {
			vec3 color = texture(halfResColor,input_uv+vec2(x,y)*halfres_invsize).rgb;
			colorSum += color;
			colorSqSum += color*color;
		}
	}
	const vec3 neighborhoodAvg = colorSum / 9.0;
	const vec3 neighborhoodVar = max(colorSqSum/9.0 - neighborhoodAvg*neighborhoodAvg, vec3(0.0));
	const vec3 neighborhoodStd = sqrt(neighborhoodVar);
	const float GAMMA = 1.0; // box half-width in stddevs; larger = looser clip, more ghosting risk
	const vec3 minColor = neighborhoodAvg - GAMMA*neighborhoodStd;
	const vec3 maxColor = neighborhoodAvg + GAMMA*neighborhoodStd;

	float blend = params.amt;
	if(params.adaptive_blend != 0) {
		float motion = clamp(length(velocity_uv_ndc) * 1000.0, 0.0, 1.0);
		blend = mix(params.stationary_blending, params.motion_blending, motion);
	}

	vec3 out_return = vec3(0.0);
	{
		vec3 clamped = clip_aabb(minColor,maxColor,neighborhoodAvg,last);
		out_return = current * (1-blend) + clamped*blend;
	}
	out_return = mix(out_return,current,weight_with_dissocclusion);

	// post-blend sharpen to counteract the temporal accumulation softening detail
	vec3 sharpen = (out_return - neighborhoodAvg) * params.sharpness;
	out_return += max(sharpen, -out_return);

	return sanitize(out_return);
}

void main()
{
	FS_OUT_Color = vec4(
		temporal_upscale(),
		1.0);
}
