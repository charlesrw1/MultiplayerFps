vec2 OctahedralEncode(vec3 normal_in)
{
	vec3 normal = normal_in.xzy;

	normal = normalize(normal);
	float l1norm = abs(normal.x) + abs(normal.y) + abs(normal.z);
	vec2 result = normal.xy * (1.0 / l1norm);
	
	if (normal.z < 0.0) {
		vec2 oldResult = result;
		result.x = (1.0 - abs(oldResult.y)) * (oldResult.x >= 0.0 ? 1.0 : -1.0);
		result.y = (1.0 - abs(oldResult.x)) * (oldResult.y >= 0.0 ? 1.0 : -1.0);
	}
	
	return result * 0.5 + 0.5; // Map to [0,1]
}


vec3 OctahedralDecode(vec2 encoded)
{
	vec2 f = encoded * 2.0 - 1.0; // Map from [0,1] to [-1,1]
	
	vec3 normal = vec3(f.x, f.y, 1.0 - (abs(f.x) + abs(f.y)));
	
	if (normal.z < 0.0) {
		vec2 oldNormal = normal.xy;
		normal.x = (1.0 - abs(oldNormal.y)) * (oldNormal.x >= 0.0 ? 1.0 : -1.0);
		normal.y = (1.0 - abs(oldNormal.x)) * (oldNormal.y >= 0.0 ? 1.0 : -1.0);
	}
	
	return normalize(normal.xzy);
}


int get_volume_index_for_global_index(int i)
{
	const int num_volumes = ddgi_global.num_volumes;
	for(int v = 1;v<num_volumes;v++) {
		if(volumes[v].size_offset.w>i) {
			return v-1;
		}
	}
	return num_volumes-1;
}


vec2 get_probe_uv_shared(vec2 octant_uv, ivec3 probe_coord, ivec3 vol_grid, int vol_offset, int DEPTH_IN_SIZE,
	int DEPTH_TILE_SIZE)
{
	const int probe_index = probe_coord.x + probe_coord.y*vol_grid.x + probe_coord.z*vol_grid.x*vol_grid.y + vol_offset;
	ivec2 tile_coords = ivec2(
		probe_index%ddgi_global.atlas_x,
		probe_index/ddgi_global.atlas_x
	);
	vec2 coords;
	coords.x = tile_coords.x*DEPTH_TILE_SIZE;
	coords.y = tile_coords.y*DEPTH_TILE_SIZE;
	
	
	coords += vec2(octant_uv*vec2(DEPTH_IN_SIZE))+vec2(1);
	vec2 final_uv = vec2(coords)/vec2(ddgi_global.atlas_x*DEPTH_TILE_SIZE,ddgi_global.atlas_y*DEPTH_TILE_SIZE);

	return final_uv;
}

vec2 get_depth_probe_uv(vec2 octant_uv, ivec3 probe_coord, ivec3 vol_grid, int vol_offset) {
	const int DEPTH_IN_SIZE = 14;
	const int DEPTH_TILE_SIZE = 16;
	return get_probe_uv_shared(octant_uv,probe_coord,vol_grid,vol_offset,DEPTH_IN_SIZE,DEPTH_TILE_SIZE);
}

vec2 get_irrad_probe_uv(vec2 octant_uv, ivec3 probe_coord, ivec3 vol_grid, int vol_offset) {
	const int DEPTH_IN_SIZE = 6;
	const int DEPTH_TILE_SIZE = 8;
	return get_probe_uv_shared(octant_uv,probe_coord,vol_grid,vol_offset,DEPTH_IN_SIZE,DEPTH_TILE_SIZE);
}

vec3 get_probe_world_from_coords(ivec3 coords, vec3 volume_origin, vec3 volume_spacing, ivec3 vol_grid, int vol_offset)
{
	
	const int global_index = coords.x + coords.y*vol_grid.x + coords.z*vol_grid.x*vol_grid.y + vol_offset;
	vec3 probe_move = ddgi_probe_relocations[global_index].xyz;
	vec3 original_pos = volume_origin+vec3(coords)*volume_spacing;

	return  original_pos+probe_move;
}
vec3 get_probe_world_from_coords_original(ivec3 coords, vec3 volume_origin, vec3 volume_spacing)
{
	vec3 original_pos = volume_origin+vec3(coords)*volume_spacing;

	return  original_pos;
}

bool is_point_in_aabb(vec3 pos, vec3 bmin, vec3 bmax)
{
    return all(greaterThanEqual(pos, bmin)) &&
           all(lessThanEqual(pos, bmax));
}

// find volume (with priority etc)
int find_best_ddgi_volume(vec3 worldpos)
{
	const int num_volumes = ddgi_global.num_volumes;
	for(int v = 0;v<num_volumes;v++) {
		vec3 volume_origin = volumes[v].origin_pri.xyz;
		vec3 density = volumes[v].density.xyz;
		ivec3 size = volumes[v].size_offset.xyz;
		
		if(is_point_in_aabb(worldpos,volume_origin,volume_origin+(size-ivec3(1))*density)) {
			return v;
		}
	}
	return num_volumes-1;

}
