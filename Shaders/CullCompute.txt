struct CullData {
	vec4 frustum_up;
	vec4 frustum_down;
	vec4 frustum_l;
	vec4 frustum_r;
	
	int num_objects;
	float inv_two_times_tanfov_2;
	float p00;
	float p11;
	
	vec4 camera_origin;
	
	float near;
	int pyramid_width;
	int pyramid_height;
	int padding2;
	
	mat4 view;
	
	// Real main camera's inv_two_times_tanfov_2 (not the shadow frustum's), so cascade
	// LOD selection matches what the main view would pick for the same instance.
	float main_view_inv_two_times_tanfov_2;
	int padding1;
	int padding3;
	int padding4;

	vec4 backplane;

	// Always the real camera position, even in shadow passes (unlike camera_origin,
	// which is the shadow frustum's own origin -- meaningless as a distance reference
	// for a directional light, which has no position). Used for the cull_distance check
	// so sun-cascade shadow casters get thinned by the same camera-relative distance as
	// the main view, instead of either not culling at all or culling by a nonsense origin.
	vec4 main_view_origin;
};
layout(binding = 5) uniform CullDataBuffer {
	CullData cull;
};

float get_screen_percentage_2(vec4 bounding_sphere, float inv_two_times_tanfov_2, float camera_dist_2)
{
	return (bounding_sphere.w * bounding_sphere.w) * inv_two_times_tanfov_2 / camera_dist_2;
}

// 2D Polyhedral Bounds of a Clipped, Perspective-Projected 3D Sphere. Michael Mara, Morgan McGuire. 2013
bool projectSphere(vec3 C, float r, float znear, float P00, float P11, out vec4 aabb)
{
	if (C.z+r <  znear)
		return false;

	vec2 cx = -C.xz;
	vec2 vx = vec2(sqrt(dot(cx, cx) - r * r), r);
	vec2 minx = mat2(vx.x, vx.y, -vx.y, vx.x) * cx;
	vec2 maxx = mat2(vx.x, -vx.y, vx.y, vx.x) * cx;

	vec2 cy = -C.yz;
	vec2 vy = vec2(sqrt(dot(cy, cy) - r * r), r);
	vec2 miny = mat2(vy.x, vy.y, -vy.y, vy.x) * cy;
	vec2 maxy = mat2(vy.x, -vy.y, vy.y, vy.x) * cy;

	aabb = vec4(minx.x / minx.y * P00, miny.x / miny.y * P11, maxx.x / maxx.y * P00, maxy.x / maxy.y * P11);
	aabb = aabb.xwzy * vec4(0.5f, -0.5f, 0.5f, -0.5f) + vec4(0.5f); // clip space -> uv space

	return true;
}
layout(binding = 0) uniform sampler2D depth_pyramid;
void main()
{
	uint gID = gl_GlobalInvocationID.x;
	// cull object id
#ifdef COMPACT_INST
	if(gID>=uint(params.compact_count))
		return;
	// Reconstruct this instance's world bounding sphere from its packed transform
	// and the batch's model-space sphere: center = pos + rot*(local_center*scale).
	CompactInstance ci = compact_instances[gID];
	CompactBatchDesc desc = compact_descs[int(ci.packed_batch_seed & 0xFFFFu)];
	mat3 ci_rot = quat_to_mat3(unpack_quat_snorm8(ci.packed_quat));
	vec3 ci_world_center = vec3(ci.pos_x, ci.pos_y, ci.pos_z) + ci_rot * (desc.local_sphere.xyz * ci.scale);
	vec4 compact_bounds = vec4(ci_world_center, desc.local_sphere.w * ci.scale);
	const int obj_flags = desc.flags;
#else
	if(gID>=cull.num_objects)
		return;

	const int obj_flags = CullObjects.data[gID].misc.w;
#endif

	#ifndef MAINVIEW
	if(!bool(obj_flags&1)) {
		return;	// not shadow caster
	}
	#endif
		
	#ifdef MAINVIEW
		// save to visarray
	uint dword_index = gID/32;	// 2bits per entry
	uint visword_shift = (gID % 32);
	if(params.second_pass != 0) {
		uint and_word = ~(1<<visword_shift);
		uint word = atomicAnd(visbits[dword_index],and_word);
		if(bool(word&(1<<visword_shift))) {
			return;	// already drawn
		}
	}
	#endif

#ifdef COMPACT_INST
	vec4 bounding_sphere = compact_bounds;
#else
	vec4 bounding_sphere = CullObjects.data[gID].bounds_sphere;
#endif

	// frustum cull (just do again on 2nd pass)
	vec3 center = bounding_sphere.xyz;
	float radius = bounding_sphere.w;
	int res = 0;
	res += int(dot(cull.frustum_up.xyz,center)+cull.frustum_up.w >= -radius);
	res += int(dot(cull.frustum_down.xyz,center)+cull.frustum_down.w >= -radius);
	res += int(dot(cull.frustum_l.xyz,center)+cull.frustum_l.w >= -radius);
	res += int(dot(cull.frustum_r.xyz,center)+cull.frustum_r.w >= -radius);
	
	#ifdef SHADOW_SPOT
	res += int(dot(cull.backplane.xyz,center)+cull.backplane.w >= -radius);
	if(res!=5)
		return;
	#else
	if(res!=4)
		return;
	#endif
	
	
#ifdef MAINVIEW
	// occlusion cull, from vkguide
	if(params.occlusion_cull != 0) {
		vec3 vs_center = (cull.view * vec4(center,1.0)).xyz;
		vs_center.y *= -1;
		vs_center.z *= -1;	// [0,+inf]
		vec4 aabb;
		//radius *= radius_bias;
		if(projectSphere(vs_center, radius, cull.near, cull.p00, cull.p11, aabb)) {
			float width = (aabb.z - aabb.x) * cull.pyramid_width;
			float height = (aabb.w - aabb.y) * cull.pyramid_height;

			//find the mipmap level that will match the screen size of the sphere
			float level = floor(log2(max(width, height)));

			//sample the depth pyramid at that specific level
			float depth = textureLod(depth_pyramid, (aabb.xy + aabb.zw) * 0.5, level).x;

			// Slack scaled to the sphere's own radius, not a flat world-unit offset --
			// a flat bias is negligible for large/far occluders (spanning many pixels,
			// where Hi-Z mip quantization error is largest) and oversized relative to
			// small/near ones, so it under-protects the cases that need it most.
			float depth_bias = radius * 0.05;
			float depthSphere = cull.near /  max((vs_center.z - radius - depth_bias),cull.near);

			if(depth>depthSphere+1e-5)	// 1e-4 -> 1e-5 epsilon
				return;	// occluded
		}
		else {
			//return;
		}
	}

	// set vis bit
	atomicOr(visbits[dword_index],1<<visword_shift); 	
#endif
	
#ifdef COMPACT_INST
	int model_ofs = desc.model_ofs;
#else
	int model_ofs = CullObjects.data[gID].misc.x;
#endif

	int num_lods = int(model_info[model_ofs]);
	float cull_distance = uintBitsToFloat(model_info[model_ofs+1]);

#if defined(MAINVIEW) || defined(SHADOW_CASCADE)
	// Real camera distance (never the light's), so sun-cascade shadow casters get
	// thinned by the same camera-relative cull distance as the main view -- otherwise
	// foliage/debris well outside render range still costs full shadow-cull work in
	// every cascade. Spot lights already cull correctly via their own frustum/backplane
	// (radius) test above, so this doesn't apply there -- see CullData::main_view_origin.
	vec3 to_main_camera = bounding_sphere.xyz - cull.main_view_origin.xyz;
	const float dist_to_main_camera_2 = dot(to_main_camera, to_main_camera);
	// 0 = never cull; otherwise drop instances beyond the asset's cull distance (meters)
	if(cull_distance>0.0 && dist_to_main_camera_2>=cull_distance*cull_distance)
		return;
#endif

	// get lod to render
#ifdef SHADOW_CASCADE
	// Same screen-percentage metric the main view uses, evaluated from the real camera's
	// distance/fov -- so a sun-cascade shadow caster renders the same LOD as the main
	// view picks for it, instead of a cascade-texel-density-based LOD that can mismatch
	// the on-screen mesh and cause shadow/silhouette seams or self-shadow acne.
	const float percentage_2 = get_screen_percentage_2(bounding_sphere, cull.main_view_inv_two_times_tanfov_2, dist_to_main_camera_2);
#else
	vec3 to_camera = bounding_sphere.xyz - cull.camera_origin.xyz;
	const float dist_to_camera_2 = dot(to_camera, to_camera);
	const float percentage_2 = get_screen_percentage_2(bounding_sphere, cull.inv_two_times_tanfov_2, dist_to_camera_2);
#endif

	int lod_wanted = 0;
	for(int i=num_lods-1;i>0;i--) {
		float end = uintBitsToFloat(model_info[model_ofs+2+(i*3)+2]);
		if(percentage_2<=end) {
			lod_wanted = i;
			break;
		}
	}
	lod_wanted = (params.force_lod>=0)?min(params.force_lod,num_lods-1) : lod_wanted;
	//lod_wanted = 0;

	int part_ofs = int(model_info[model_ofs+2+(lod_wanted*3)]);
	int part_count = int(model_info[model_ofs+2+(lod_wanted*3)+1]);

#ifdef COMPACT_INST
	int mat_override = desc.mat_ofs;
#else
	int mat_override = CullObjects.data[gID].misc.z;
#endif
	for(int i=0;i<part_count;i++) {
		int command_index = int(model_info[model_ofs+2+num_lods*3 + (part_ofs + i)*2]);
		int mat_offset = int(model_info[model_ofs+2+num_lods*3 + (part_ofs + i)*2 + 1]);
		
		//int command_index = int(model_info[model_ofs+1+num_lods*3+part_index]);
		
	
		uint count_index = atomicAdd(draw_cmd_out[command_index].primCount,1);
		uint instanceIndex = draw_cmd_out[command_index].baseInstance + count_index;
#ifdef COMPACT_INST
		object_inst_indirect[instanceIndex*2] = gID;	// compact instance's own dense index
#else
		object_inst_indirect[instanceIndex*2] = CullObjects.data[gID].misc.y;	// the index of object
#endif
		
		mat_offset = (mat_override>0)?mat_override:mat_offset;
		
		object_inst_indirect[instanceIndex*2 + 1] = mat_offset;
		
	
	}
}
