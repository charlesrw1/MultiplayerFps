
float unlinearize_depth(float linear)
{
	return g.near / linear;
}
float linearize_depth(float depth)
{
	return g.near / depth;
}

float get_world_depth_from_volume_normalized(float normalized_z, float spread, float size)
{
	float a = pow(normalized_z,spread);
	float tile = a * size + g.near;
	return tile;
}
float get_volume_normalized_from_depth(float world_depth, float spread, float size)
{
	return pow((world_depth-g.near)/size, 1.0/spread);
}

vec4 sample_volfog_volume(vec3 NDC, sampler3D fog_volume)
{
	float lineardepth = linearize_depth(NDC.z);
	vec3 texturecoord = vec3((NDC.xy+vec2(1.0))*0.5,0.0);
	
	float normalized_z = get_volume_normalized_from_depth(lineardepth,fogparam.spread_frustumend.x,fogparam.spread_frustumend.y);
	texturecoord.z = normalized_z;

	return texture(fog_volume,texturecoord);
}
