
"BaseEntity" abstract {
	vec3 position "0 0 0"
	vec3 scale "1 1 1"
	quat rotation "1 0 0 0"
	color32 color "255 255 255 255"
	model model ""

	SIGNAL OnKill : "(void) called when entity is removed"
	EVENT Activate : "(void) activate an entity from dormancy"
}

"StaticMesh" {
	inherits BaseEntity
	bool cast_shadow "1"
	bool has_collisions "1"
}

"PhysicsProp" {
	inherits BaseEntity
}

"NPC" : "a default NPC" {
	inherits BaseEntity
	_edimage "icon/smile.png"
	
	; dont expose this to editor
	cppclass _classname "NPC"
	
	bool can_use_item "0" : "can player use item"
	int attribute_something "0,0,5" : "something"
	
	SET model "boat.cmdl"
	
	SIGNAL Fire : "(void) Callled when NPC fires weapon"
	SIGNAL Killed : "(void) Called when NPC was killed"
	SIGNAL FindTarget : "(entityhandle) Called when NPC finds target"
	
	EVENT SetIdle : "Sets NPC to idle"
	EVENT Out : "Sets NPC output"
}

"PlayerSpawn" {
	inherits BaseEntity
	_edmodel player_FINAL.cmdl
}



"Trigger" {
	inherits BaseEntity
	_edmodel spheres.cmdl
	cppclass _classname "Trigger"
	
	SIGNAL OnEnter : "(void) called when something enters trigger (!activator)"
	SIGNAL OnExit : "(void) called when something leaves trigger (!activator)"
}
